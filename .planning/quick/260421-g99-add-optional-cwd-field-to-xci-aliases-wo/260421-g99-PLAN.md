<objective>
Add optional `cwd` field to every xci alias kind (single, sequential, parallel, for_each, ini). An alias declaring `cwd: <path>` runs its command(s) with that working directory instead of `projectRoot`. Relative paths resolve against `projectRoot`; absolute paths are used verbatim; `${placeholder}` strings are interpolated at resolve time. A child alias inherits the parent's cwd when it doesn't declare its own.

Purpose: Monorepo users want per-alias working directories (e.g. `packages/auth` build runs from its own folder) without shelling through `cd` tricks that break cross-platform. Today every command runs in `projectRoot` — users work around this by embedding `cd` in the command, which fails on Windows cmd.exe with shell:false.

Output: `cwd` is a first-class field in CommandDef and ExecutionPlan; executors respect it per-step; resolver threads it through alias inheritance; CLI resolves relative→absolute; normalize validates type; params.ts tracks ${...} placeholders inside it; run-header / dry-run / verbose show it with secret redaction; README documents it.
</objective>

## 2.1 — Create packages/xci/src/executor/cwd.ts (new file)

```ts
// src/executor/cwd.ts
//
// Resolve every cwd field on an ExecutionPlan to an absolute path against projectRoot.
// Relative paths become projectRoot-relative. Absolute paths pass through. Undefined stays undefined.
// Called by cli.ts between resolver.resolve() and executor.run().

import { isAbsolute, resolve as resolvePath } from 'node:path';
import type { ExecutionPlan, SequentialStep } from '../types.js';

function toAbs(cwd: string | undefined, projectRoot: string): string | undefined {
  if (cwd === undefined) return undefined;
  if (isAbsolute(cwd)) return cwd;
  return resolvePath(projectRoot, cwd);
}

function resolveStepCwd(step: SequentialStep, projectRoot: string): SequentialStep {
  // 'set' steps never have cwd — pass through unchanged.
  if (step.kind === 'set') return step;
  // Both 'ini' and the cmd variant (kind undefined or 'cmd') may carry cwd.
  const abs = toAbs(step.cwd, projectRoot);
  if (abs === undefined) {
    // Strip the possibly-relative undefined-case — but since undefined stays undefined, just return step.
    return step;
  }
  return { ...step, cwd: abs };
}

export function resolveAbsoluteCwds(plan: ExecutionPlan, projectRoot: string): ExecutionPlan {
  switch (plan.kind) {
    case 'single': {
      const abs = toAbs(plan.cwd, projectRoot);
      return abs === undefined ? plan : { ...plan, cwd: abs };
    }
    case 'sequential':
      return { ...plan, steps: plan.steps.map((s) => resolveStepCwd(s, projectRoot)) };
    case 'parallel':
      return {
        ...plan,
        group: plan.group.map((entry) => {
          const abs = toAbs(entry.cwd, projectRoot);
          return abs === undefined ? entry : { ...entry, cwd: abs };
        }),
      };
    case 'ini': {
      const abs = toAbs(plan.cwd, projectRoot);
      return abs === undefined ? plan : { ...plan, cwd: abs };
    }
  }
}
```
