<objective>
Make transactional email links (verify-email, password-reset, org invite) use a configurable `APP_BASE_URL` env var, falling back to the current `https://${req.headers.host}` construction when the env var is absent. This fixes broken `https://localhost:3000/...` links emitted by the dev docker-compose stack while preserving backward compatibility for existing prod deployments that rely on the Host header.

Purpose: Dev stack emits clickable links out of the box; prod operators can set an explicit canonical base URL.
Output: One helper module, one test file, three route call-site updates, two env-schema additions, two config-surface edits (docker-compose.yml, .env.example).
</objective>

```typescript
export interface BuildEmailLinkCtx {
  // fastify.config.APP_BASE_URL — optional; verbatim prefix when set
  appBaseUrl: string | undefined;
  // req.headers.host — used only as fallback
  headerHost: string | undefined;

export function buildEmailLink(
  ctx: BuildEmailLinkCtx,
  path: string,      // must start with '/', e.g. '/verify-email'
  queryKey: string,  // e.g. 'token'
  queryValue: string // raw token, helper encodes it
): string;
```
