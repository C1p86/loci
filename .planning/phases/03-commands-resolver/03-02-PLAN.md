export type ExecutionPlan =
  | { readonly kind: 'single'; readonly argv: readonly string[] }
  | { readonly kind: 'sequential'; readonly steps: readonly (readonly string[])[] }

      readonly kind: 'parallel';
      readonly group: readonly {
        readonly alias: string;
        readonly argv: readonly string[];
      }[];

  resolve(aliasName: string, commands: CommandMap, config: ResolvedConfig): ExecutionPlan;

  | { readonly kind: 'single'; readonly cmd: readonly string[]; readonly description?: string; readonly platforms?: PlatformOverrides; }
  | { readonly kind: 'sequential'; readonly steps: readonly CommandRef[]; readonly description?: string; }

export type CommandMap = ReadonlyMap<string, CommandDef>;

    Internal function `resolveAlias(aliasName: string, commands: CommandMap, config: ResolvedConfig, depth: number, chain: string[]): ExecutionPlan`:

    - If `depth > 10`: throw `new CommandSchemaError(chain[0], 'alias nesting exceeds maximum depth of 10: ' + chain.join(' -> '))`
    - Get def from commands.get(aliasName). If undefined: throw `new UnknownAliasError(aliasName)`
    - Switch on `def.kind`:

    **'single':**
    - `const cmd = selectPlatformCmd(def, aliasName)`
    - `const argv = interpolateArgv(cmd, aliasName, config.values)`
    - Return `{ kind: 'single', argv }`

    **'sequential':**
    - For each step in `def.steps`:
      - If `commands.has(step)` (D-09: lookup-based alias detection):
        - Recursively resolve: `resolveAlias(step, commands, config, depth + 1, [...chain, step])`
        - Flatten: if the recursive result is 'single', take its argv. If 'sequential', spread its steps. If 'parallel', wrap as a single step (the parallel group becomes one step in the sequence -- but this is an edge case; for simplicity, flatten only 'single' results and embed 'sequential' steps inline).
        - Actually, the ExecutionPlan for sequential expects `steps: readonly (readonly string[])[]` -- each step is a flat argv. So recursive resolution of a 'sequential' sub-alias should merge its steps into the parent. A 'parallel' sub-alias cannot be flattened into sequential steps -- it should remain as-is. But the type constraint says steps are argv arrays. Solution: expand nested sequential into flat steps. For nested parallel or nested sequential-containing-parallel, this is a Phase 4 concern. For Phase 3, resolve each step:
        - If the referenced alias is 'single': resolve to argv, add as one step
        - If the referenced alias is 'sequential': resolve each of its sub-steps recursively, add all as separate steps
        - If the referenced alias is 'parallel': this is a type mismatch for sequential embedding. For now, resolve the parallel alias and embed it as a single entry -- but ExecutionPlan.sequential.steps expects argv arrays. This means a parallel alias embedded in a sequential alias is resolved by Phase 4 running the parallel group as one step. However, the current type does not support this.
        - SIMPLIFICATION: Per the locked ExecutionPlan type, sequential steps are flat argv arrays. A step that references another sequential alias expands inline (its steps merge). A step that references a parallel alias is an error at this level -- the user should restructure their commands. OR: resolve each step to a single argv (if single), or expand steps (if sequential). If parallel: throw CommandSchemaError explaining that parallel aliases cannot be embedded in sequential steps.
        - Actually re-reading the types more carefully: `ExecutionPlan.sequential.steps` is `readonly (readonly string[])[]` -- just arrays of argv. So each step must resolve to a single argv. Recursive sequential expands (multiple argvs). Recursive parallel cannot flatten to a single argv.
        - For D-09 inline commands: if the step does NOT match a CommandMap key, it's an inline command string. Tokenize it with `tokenize(step, aliasName)`, then interpolate with `interpolateArgv(tokens, aliasName, config.values)`. Add the resulting argv as one step.
      - Else (inline command per D-09):
        - `const tokens = tokenize(step, aliasName)`
        - `const argv = interpolateArgv(tokens, aliasName, config.values)`
        - Add argv as one step
    - Return `{ kind: 'sequential', steps: allSteps }`

    **'parallel':**
    - For each entry in `def.group`:
      - If `commands.has(entry)`:
        - Resolve recursively. If result is 'single': use `{ alias: entry, argv: result.argv }`
        - If result is 'sequential' or 'parallel': for parallel group type, each entry expects `{ alias, argv }`. A nested sequential/parallel in a parallel group is complex. SIMPLIFICATION: only 'single' aliases can be resolved directly into parallel group entries. For 'sequential' sub-aliases in parallel groups, throw CommandSchemaError. Actually, the parallel group type says `readonly { alias: string; argv: readonly string[] }[]` -- each entry is one command. So only single-command aliases can be parallel group entries. Sequential sub-aliases embedded in parallel would need special handling (run the sequence as one unit). This is Phase 4's executor responsibility.
        - REVISED APPROACH: For parallel groups, each entry resolves to one `{ alias, argv }`. If the referenced alias is 'single', straightforward. If 'sequential', this is the executor's job to run the sequence -- but we need to represent it. Since ExecutionPlan.parallel.group expects `{ alias, argv }` (single command), we cannot represent a sequential sub-alias here.
        - PRAGMATIC SOLUTION: At the resolver level, if a parallel group entry references a non-single alias, throw `CommandSchemaError(aliasName, 'parallel group entry "${entry}" must resolve to a single command')`. This is consistent with the locked type constraint. Users compose aliases at the outer level.

        - `const argv = interpolateArgv(tokens, aliasName, config.values)`
        - Add `{ alias: entry, argv }` to group
    - Return `{ kind: 'parallel', group: allEntries }`

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-05 | Information Disclosure | interpolate.ts | mitigate | Interpolation operates on argv tokens, not shell strings. UndefinedPlaceholderError receives only the key name, never the value. No config value appears in any error message. |
| T-03-06 | Information Disclosure | envvars.ts redaction | mitigate | redactSecrets() replaces secret env values with `***` using the secretKeys set. Real values only flow to Phase 4's actual env injection (12-factor compliant per D-07). |
| T-03-07 | Tampering | interpolate.ts command injection | mitigate | Interpolated values stay within their argv token (INT-03). No re-splitting of tokens after interpolation. Phase 4 uses shell:false (EXE-01), so metacharacters in interpolated values are inert. |
| T-03-08 | Denial of Service | resolver depth cap | mitigate | Alias nesting depth enforced at 10 (D-10). resolveAlias tracks depth counter and chain; exceeding 10 throws CommandSchemaError with full chain for debugging. |
| T-03-09 | Information Disclosure | error messages in resolver | mitigate | All error constructors receive only alias names, placeholder key names, and structural descriptions. Never config values. Follows ShellInjectionError `void value` precedent. |
