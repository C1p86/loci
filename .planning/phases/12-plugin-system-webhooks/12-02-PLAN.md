```ts
import type { FastifyRequest } from 'fastify';

export interface VerifySuccess { ok: true; deliveryId: string; }
export interface VerifyFailure { ok: false; reason: 'signature_missing' | 'signature_mismatch' | 'token_mismatch' | 'header_missing'; }
export type VerifyResult = VerifySuccess | VerifyFailure;

export interface GitHubPushEvent {
  kind: 'push';
  ref: string;
  repository: string;         // repository.full_name e.g. 'acme/infra'
  sha: string;                // head_commit.id (empty string if deleted)
  pusher: string;             // pusher.name
  message: string;            // head_commit.message
}
export interface GitHubPullRequestEvent {
  kind: 'pull_request';
  action: 'opened' | 'synchronize' | 'closed' | 'reopened';
  repository: string;
  number: number;
  headRef: string;
  baseRef: string;
  title: string;
}
export type GitHubEvent = GitHubPushEvent | GitHubPullRequestEvent;

export interface PerforceEvent {
  kind: 'perforce_change';
  change: string;
  user: string;
  client: string;
  root: string;
  depot: string;
  deliveryId: string;  // UUID generated by the xci-emit script (D-24)
}

export interface GitHubTriggerConfig {
  plugin: 'github';
  events: Array<'push' | 'pull_request'>;
  repository?: string;
  branch?: string;
  actions?: Array<'opened' | 'synchronize' | 'closed' | 'reopened'>;
}
export interface PerforceTriggerConfig {
  plugin: 'perforce';
  depot?: string;
  user?: string;
  client?: string;
}
export type TriggerConfig = GitHubTriggerConfig | PerforceTriggerConfig;

export interface TaskTriggerMatch {
  taskId: string;
  params: Record<string, string>;
}

/**
 * The plugin ownership of each event type stays inside the concrete plugin module.
 * Signed with generics so the shared route handler does not couple to either event.
 */
export interface TriggerPlugin<E = unknown> {
  name: 'github' | 'perforce';
  /**
   * Verifies the request's authenticity using the provided pluginSecret (GitHub HMAC key,
   * or for Perforce: the decoded plaintext X-Xci-Token expectation — passed as Buffer).
   * pluginSecret === null means the plugin stores no secret (Perforce verifies via URL token
   * match at the route layer; plugin-level verify still checks X-Xci-Token header present).
   */
  verify(req: FastifyRequest, pluginSecret: Buffer | null): VerifyResult;
  /**
   * Parses req.body + req.headers into a typed event. Returns null for supported-but-ignored
   * events (GitHub 'issues', 'workflow_run' etc.). Throws on malformed body.
   */
  parse(req: FastifyRequest): E | null;
  /**
   * Matches the event against the array of trigger_configs (from every task in the org).
   * Returns ALL matches; one event can fire multiple tasks.
   * The [taskId, taskTriggerConfigs] tuple supplies which task each config originated from
   * (the caller builds this list by joining tasks with their trigger_configs).
   */
  mapToTask(
    event: E,
    candidates: Array<{ taskId: string; configs: TriggerConfig[] }>,
  ): TaskTriggerMatch[];
}
```

```ts
/**
 * Claude's discretion per D-36 — hand-rolled `*` wildcard match.
 * `*` matches one or more chars (greedy, any chars including `/` for Perforce depot compat).
 * Other regex metachars in the pattern are literal-escaped.
 * D-10 glob matching rules: 'acme/*' matches 'acme/infra'; 'main' is literal; '*' matches all.
 */
export function matchGlob(pattern: string, value: string): boolean {
  // Escape regex metachars EXCEPT '*', which we turn into '.+'
  const re =
    '^' +
    pattern.split('*').map((p) => p.replace(/[.+?^${}()|[\]\\]/g, '\\$&')).join('.+') +
    '$';
  return new RegExp(re).test(value);
}
```

If `pattern === '*'`, `re === '^.+$'` — matches any non-empty string. Empty pattern matches only empty string. Test this exhaustively.

```ts
import { createHmac } from 'node:crypto';
import type { FastifyRequest } from 'fastify';
import { compareToken } from '../crypto/tokens.js';
import { matchGlob } from './glob.js';
import type {
  GitHubEvent,
  GitHubTriggerConfig,
  TriggerConfig,
  TriggerPlugin,
  TaskTriggerMatch,
  VerifyResult,
} from './types.js';

/**
 * GitHub webhook plugin — PLUG-03.
 * D-08: HMAC-SHA256 via X-Hub-Signature-256 header, timingSafeEqual compare.
 * D-09: 'push' + 'pull_request' events only; others return null from parse().
 * D-11: param extraction populates git.* (push) or pr.* (PR) keys.
 */
const githubPlugin: TriggerPlugin<GitHubEvent> = {
  name: 'github',

  verify(req, pluginSecret) {
    if (pluginSecret === null) {
      // GitHub plugin requires a secret — stored at webhook_tokens creation time (D-28).
      return { ok: false, reason: 'header_missing' };
    }
    const deliveryId = req.headers['x-github-delivery'];
    if (typeof deliveryId !== 'string' || deliveryId.length === 0) {
      return { ok: false, reason: 'header_missing' };
    }
    const sigHeader = req.headers['x-hub-signature-256'];
    if (typeof sigHeader !== 'string' || !sigHeader.startsWith('sha256=')) {
      return { ok: false, reason: 'signature_missing' };
    }
    const providedHex = sigHeader.slice('sha256='.length);

    // req.rawBody must be populated by the route handler via Fastify onRequest or
    // a dedicated contentTypeParser that captures raw bytes before JSON.parse.
    // Plan 12-03 wires this: see shared-handler preHandler.
    const raw = (req as unknown as { rawBody?: Buffer }).rawBody;
    if (!raw) {
      // Should never happen if the route handler sets rawBody; treat as mismatch rather than crash.
      return { ok: false, reason: 'signature_mismatch' };
    }

    const computedHex = createHmac('sha256', pluginSecret).update(raw).digest('hex');
    if (!compareToken(providedHex, computedHex)) {
      return { ok: false, reason: 'signature_mismatch' };
    }
    return { ok: true, deliveryId };
  },

  parse(req) {
    const event = req.headers['x-github-event'];
    if (typeof event !== 'string') return null;
    // biome-ignore lint/suspicious/noExplicitAny: webhook payload has dynamic shape
    const body = req.body as any;

    if (event === 'push') {
      // Defensive: GitHub always sends ref + repository.full_name; guard anyway.
      if (typeof body?.ref !== 'string' || typeof body?.repository?.full_name !== 'string') {
        throw new Error('github_push_malformed');
      }
      return {
        kind: 'push',
        ref: body.ref,
        repository: body.repository.full_name,
        sha: body.head_commit?.id ?? '',
        pusher: body.pusher?.name ?? '',
        message: body.head_commit?.message ?? '',
      };
    }

    if (event === 'pull_request') {
      if (
        typeof body?.action !== 'string' ||
        typeof body?.pull_request?.head?.ref !== 'string' ||
        typeof body?.repository?.full_name !== 'string'
      ) {
        throw new Error('github_pr_malformed');
      }
      const action = body.action;
      if (!['opened', 'synchronize', 'closed', 'reopened'].includes(action)) {
        // Unknown action — treat as ignored event (null) rather than throwing
        return null;
      }
      return {
        kind: 'pull_request',
        action,
        repository: body.repository.full_name,
        number: body.pull_request.number,
        headRef: body.pull_request.head.ref,
        baseRef: body.pull_request.base.ref,
        title: body.pull_request.title ?? '',
      };
    }

    return null; // D-09: ignored events (issues, workflow_run, etc.)
  },

  mapToTask(event, candidates) {
    const matches: TaskTriggerMatch[] = [];
    for (const { taskId, configs } of candidates) {
      for (const cfg of configs) {
        if (cfg.plugin !== 'github') continue;
        const g = cfg as GitHubTriggerConfig;
        if (!g.events.includes(event.kind)) continue;
        if (g.repository && !matchGlob(g.repository, event.repository)) continue;

        if (event.kind === 'push') {
          // branch = ref after 'refs/heads/'; empty for tag or non-branch refs
          const branch = event.ref.startsWith('refs/heads/')
            ? event.ref.slice('refs/heads/'.length)
            : '';
          if (g.branch && !matchGlob(g.branch, branch)) continue;
          matches.push({
            taskId,
            params: {
              'git.ref': event.ref,
              'git.sha': event.sha,
              'git.repository': event.repository,
              'git.pusher': event.pusher,
              'git.message': event.message,
            },
          });
        } else {
          // pull_request
          if (g.actions && g.actions.length > 0 && !g.actions.includes(event.action)) continue;
          if (g.branch && !matchGlob(g.branch, event.headRef)) continue;
          matches.push({
            taskId,
            params: {
              'pr.number': String(event.number),
              'pr.action': event.action,
              'pr.head_ref': event.headRef,
              'pr.base_ref': event.baseRef,
              'pr.title': event.title,
              'git.repository': event.repository,
            },
          });
        }
        // One config per task is enough — break inner loop to avoid duplicate entries
        break;
      }
    }
    return matches;
  },
};

export default githubPlugin;
```

```ts
import { randomUUID } from 'node:crypto';
import type { FastifyRequest } from 'fastify';
import { compareToken } from '../crypto/tokens.js';
import { matchGlob } from './glob.js';
import type {
  PerforceEvent,
  PerforceTriggerConfig,
  TriggerPlugin,
  TaskTriggerMatch,
  VerifyResult,
} from './types.js';

/**
 * Perforce webhook plugin — PLUG-04.
 * D-13: verify via X-Xci-Token header (no HMAC; Perforce trigger scripts can't easily do it in shell).
 * D-14: body is {change, user, client, root, depot} JSON posted by xci-emit-perforce-trigger script.
 * pluginSecret here is the plaintext EXPECTED token bytes (Buffer) stored encrypted at rest per D-28.
 * However, for Perforce, the actual verification uses the URL orgToken that already identified the org —
 * the pluginSecret parameter for this plugin's verify() is IGNORED at the route layer in favor of
 * header-based double-check. Plan 12-03 passes pluginSecret=null; the route handler verifies the URL
 * token separately via adminRepo.findWebhookTokenByPlaintext. This plugin's verify ensures:
 *   - X-Xci-Token header is PRESENT (defense in depth against misconfigured proxies stripping URL)
 *   - body.delivery_id is present OR auto-generated (D-24)
 */
const perforcePlugin: TriggerPlugin<PerforceEvent> = {
  name: 'perforce',

  verify(req, _pluginSecret) {
    const xciToken = req.headers['x-xci-token'];
    if (typeof xciToken !== 'string' || xciToken.length === 0) {
      return { ok: false, reason: 'header_missing' };
    }
    // biome-ignore lint/suspicious/noExplicitAny: body parsed by Fastify JSON parser
    const body = req.body as any;
    const deliveryId =
      typeof body?.delivery_id === 'string' && body.delivery_id.length > 0
        ? body.delivery_id
        : randomUUID();
    return { ok: true, deliveryId };
  },

  parse(req) {
    // biome-ignore lint/suspicious/noExplicitAny: JSON body
    const body = req.body as any;
    if (!body || typeof body !== 'object') {
      throw new Error('perforce_body_not_object');
    }
    const { change, user, client, root, depot } = body;
    if (
      typeof change !== 'string' ||
      typeof user !== 'string' ||
      typeof client !== 'string' ||
      typeof root !== 'string' ||
      typeof depot !== 'string'
    ) {
      throw new Error('perforce_body_missing_fields');
    }
    const deliveryId =
      typeof body.delivery_id === 'string' && body.delivery_id.length > 0
        ? body.delivery_id
        : randomUUID();
    return { kind: 'perforce_change', change, user, client, root, depot, deliveryId };
  },

  mapToTask(event, candidates) {
    const matches: TaskTriggerMatch[] = [];
    for (const { taskId, configs } of candidates) {
      for (const cfg of configs) {
        if (cfg.plugin !== 'perforce') continue;
        const p = cfg as PerforceTriggerConfig;
        if (p.depot && !matchGlob(p.depot, event.depot)) continue;
        if (p.user && !matchGlob(p.user, event.user)) continue;
        if (p.client && !matchGlob(p.client, event.client)) continue;
        matches.push({
          taskId,
          params: {
            'p4.change': event.change,
            'p4.user': event.user,
            'p4.client': event.client,
            'p4.root': event.root,
            'p4.depot': event.depot,
          },
        });
        break;
      }
    }
    return matches;
  },
};

export default perforcePlugin;
```
