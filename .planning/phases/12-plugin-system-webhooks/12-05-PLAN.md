    .description('Emit Perforce change-commit trigger scripts (sh/bat/ps1)')
    .argument('<url>', 'Webhook URL, e.g. https://xci.example.com/hooks/perforce/<orgToken>')

      process.stdout.write(`Generated ${result.files.length} scripts in ${opts.output}:\n`);

      process.stdout.write('\nIMPORTANT: These scripts contain the X-Xci-Token. Restrict file permissions on the Perforce server:\n');

    });
}

import { mkdirSync, writeFileSync } from 'node:fs';
import { join, resolve } from 'node:path';

export interface EmitOptions {
  url: string;       // webhook endpoint, e.g. https://xci.example.com/hooks/perforce/xci_whk_abc
  token: string;     // X-Xci-Token value (the orgToken plaintext)
  outputDir: string; // target directory for the 3 files
}

export interface EmitResult {
  files: readonly string[]; // relative filenames written
}

const HEADER_COMMENT_SH = (url: string) => `#!/bin/sh
#
# xci Perforce change-commit trigger вЂ” POSIX shell
# Generated by: xci agent-emit-perforce-trigger
# Endpoint: ${url}
#
# Install as a Perforce trigger:
#   p4 triggers
#   change-commit trigger command: "/path/to/trigger.sh %change% %user% %client%"
#
# SECURITY: this script contains the webhook token. Restrict file permissions:
#   chmod 700 \${0}
#   chown \${P4_SERVICE_USER} \${0}
`;

const HEADER_COMMENT_BAT = (url: string) => `@echo off
REM xci Perforce change-commit trigger - Windows cmd wrapper
REM Generated by: xci agent-emit-perforce-trigger
REM Endpoint: ${url}
REM
REM Uses PowerShell under the hood (cmd has no native HTTP client).
REM SECURITY: restrict file permissions on the Perforce server.
`;

const HEADER_COMMENT_PS1 = (url: string) => `# xci Perforce change-commit trigger - PowerShell
# Generated by: xci agent-emit-perforce-trigger
# Endpoint: ${url}
#
# Install as a Perforce trigger on Windows:
#   p4 triggers
#   change-commit trigger command: "powershell -ExecutionPolicy Bypass -File C:\\path\\to\\trigger.ps1 %change% %user% %client%"
#
# SECURITY: restrict file permissions (icacls) and store outside VCS-tracked directories.
`;

export function buildShTemplate(url: string, token: string): string {
  // JSON body uses shell var expansion in double quotes вЂ” fields unset в†’ empty string (OK for webhook)
  return `${HEADER_COMMENT_SH(url)}
set -e

# Generate a UUID for delivery_id (for idempotency dedup per D-22)
if [ -r /proc/sys/kernel/random/uuid ]; then
  DELIVERY_ID=$(cat /proc/sys/kernel/random/uuid)
elif command -v uuidgen >/dev/null 2>&1; then
  DELIVERY_ID=$(uuidgen)
else
  DELIVERY_ID="$(date +%s)-$$"
fi

# Perforce change-commit environment variables
CHANGE="\${P4_CHANGE:-\${1:-}}"
USER="\${P4_USER:-\${2:-}}"
CLIENT="\${P4_CLIENT:-\${3:-}}"
ROOT="\${P4_ROOT:-}"
DEPOT="\${P4_DEPOT_PATH:-}"

BODY="{\\"change\\":\\"\${CHANGE}\\",\\"user\\":\\"\${USER}\\",\\"client\\":\\"\${CLIENT}\\",\\"root\\":\\"\${ROOT}\\",\\"depot\\":\\"\${DEPOT}\\",\\"delivery_id\\":\\"\${DELIVERY_ID}\\"}"

curl --fail --silent --show-error \\
  -X POST \\
  -H "Content-Type: application/json" \\
  -H "X-Xci-Token: ${token}" \\
  --data-raw "\${BODY}" \\
  "${url}"
`;
}

export function buildPs1Template(url: string, token: string): string {
  return `${HEADER_COMMENT_PS1(url)}
$ErrorActionPreference = 'Stop'

# Perforce change-commit environment variables (fallback to positional args)
$change = if ($env:P4_CHANGE) { $env:P4_CHANGE } elseif ($args.Count -ge 1) { $args[0] } else { '' }
$user   = if ($env:P4_USER)   { $env:P4_USER }   elseif ($args.Count -ge 2) { $args[1] } else { '' }
$client = if ($env:P4_CLIENT) { $env:P4_CLIENT } elseif ($args.Count -ge 3) { $args[2] } else { '' }
$root   = if ($env:P4_ROOT)   { $env:P4_ROOT }   else { '' }
$depot  = if ($env:P4_DEPOT_PATH) { $env:P4_DEPOT_PATH } else { '' }
$deliveryId = [guid]::NewGuid().ToString()

$body = @{
  change     = $change
  user       = $user
  client     = $client
  root       = $root
  depot      = $depot
  delivery_id = $deliveryId
} | ConvertTo-Json -Compress

Invoke-WebRequest -Uri '${url}' -Method POST \\
  -Headers @{ 'X-Xci-Token' = '${token}'; 'Content-Type' = 'application/json' } \\
  -Body $body \\
  -UseBasicParsing | Out-Null
`;
}

export function buildBatTemplate(url: string, token: string): string {
  // .bat delegates to PowerShell; inline PS preserves security header + JSON body
  return `${HEADER_COMMENT_BAT(url)}
setlocal
set "CHANGE=%P4_CHANGE%"
if "%CHANGE%"=="" set "CHANGE=%~1"
set "P4USER_=%P4_USER%"
if "%P4USER_%"=="" set "P4USER_=%~2"
set "CLIENT=%P4_CLIENT%"
if "%CLIENT%"=="" set "CLIENT=%~3"
set "ROOT=%P4_ROOT%"
set "DEPOT=%P4_DEPOT_PATH%"

powershell -NoProfile -ExecutionPolicy Bypass -Command ^
  "$body = @{ change='%CHANGE%'; user='%P4USER_%'; client='%CLIENT%'; root='%ROOT%'; depot='%DEPOT%'; delivery_id=[guid]::NewGuid().ToString() } | ConvertTo-Json -Compress;" ^
  "Invoke-WebRequest -Uri '${url}' -Method POST -Headers @{ 'X-Xci-Token'='${token}'; 'Content-Type'='application/json' } -Body $body -UseBasicParsing | Out-Null"

endlocal
`;
}

export function emitPerforceTriggerScripts(opts: EmitOptions): EmitResult {
  const out = resolve(opts.outputDir);
  mkdirSync(out, { recursive: true });

  const sh = buildShTemplate(opts.url, opts.token);
  const ps1 = buildPs1Template(opts.url, opts.token);
  const bat = buildBatTemplate(opts.url, opts.token);

  writeFileSync(join(out, 'trigger.sh'), sh, { mode: 0o700 });
  writeFileSync(join(out, 'trigger.bat'), bat, { mode: 0o700 });
  writeFileSync(join(out, 'trigger.ps1'), ps1, { mode: 0o700 });

  return { files: ['trigger.sh', 'trigger.bat', 'trigger.ps1'] };
}
```

**Pitfall:** Node `writeFileSync` `mode: 0o700` works on Unix only; on Windows the mode is effectively ignored (NTFS uses ACLs). The admin-responsibility comment in the script header covers this вЂ” the xci emit is not a security boundary; the Perforce admin is responsible for ACLing the output.

## Plugin System & Webhooks (Phase 12)

xci server supports incoming webhooks via a pluggable trigger architecture. Two bundled plugins ship with every release: GitHub and Perforce. Plugins are bundled at build time вЂ” NO dynamic runtime install (PLUG-02).

### Webhook endpoints

- `POST /hooks/github/:orgToken` вЂ” GitHub push + pull_request events
- `POST /hooks/perforce/:orgToken` вЂ” Perforce change-commit trigger (JSON body)

Both endpoints are unauthenticated at the session layer вЂ” the `:orgToken` URL path segment identifies the org (via sha256 hash lookup in `webhook_tokens`). GitHub requests are verified by HMAC-SHA256 against the per-token `plugin_secret`. Perforce requests are verified by `X-Xci-Token` header match.

### Managing webhook tokens

`POST /api/orgs/:orgId/webhook-tokens` (Owner/Member, CSRF):
- body: `{ pluginName: 'github' | 'perforce', pluginSecret?: string }`
- returns: `{ id, plaintext, endpointUrl }` вЂ” **plaintext is returned ONCE; store it immediately**

Other endpoints: `GET /api/orgs/:orgId/webhook-tokens` (list), `POST /api/orgs/:orgId/webhook-tokens/:id/revoke`, `DELETE /api/orgs/:orgId/webhook-tokens/:id` (Owner-only).

### Dead Letter Queue (DLQ)

Any webhook that fails the verifyв†’parseв†’mapToTask pipeline lands in `dlq_entries`. Sensitive headers (`Authorization`, `X-Hub-Signature`, `X-Hub-Signature-256`, `X-GitHub-Token`, `X-Xci-Token`, `Cookie`, `Set-Cookie`) are stripped before persist (PLUG-08).

- `GET /api/orgs/:orgId/dlq` вЂ” list DLQ entries (any member; filter by plugin/reason/since)
- `POST /api/orgs/:orgId/dlq/:dlqId/retry` вЂ” replay through parseв†’mapToTaskв†’dispatch (Owner/Member; CSRF)

**Retry semantics:** signature verification is **skipped** on retry (D-20) вЂ” the admin is consciously accepting the event. The retry endpoint logs `dlq_retry_skipping_signature_verify` at WARN level for audit.

### Idempotency

The `webhook_deliveries` table tracks `(plugin_name, delivery_id)` uniquely. A duplicate GitHub `X-GitHub-Delivery` or Perforce `delivery_id` returns `200 {status:'duplicate'}` with a WARN log; no second task_run is created.

### Task trigger configuration

Tasks store `trigger_configs` JSONB on the `tasks` table вЂ” an array of `GitHubTriggerConfig | PerforceTriggerConfig`. Explicit per-task configuration (no naming convention): a task with an empty `trigger_configs` array is NOT triggerable via webhook. Validation happens on task create/update via `validateTriggerConfigs`.
```
