```typescript
// packages/xci/src/agent/runner.ts
import { execa } from 'execa';
import { execSync } from 'node:child_process';

export interface RunnerOptions {
  argv: readonly string[];
  cwd: string;
  env: Record<string, string>;
  onChunk: (stream: 'stdout'|'stderr', data: string, seq: number) => void;
  onExit: (exitCode: number, durationMs: number) => void;
}

export interface RunHandle {
  cancel: () => Promise<void>;
  runId: string;
}

export function spawnTask(runId: string, opts: RunnerOptions): RunHandle {
  const [cmd, ...args] = opts.argv;
  if (!cmd) throw new Error('empty argv');
  const startTime = Date.now();
  let seq = 0;
  const proc = execa(cmd, args, {
    cwd: opts.cwd,
    env: {...process.env, ...opts.env},
    stdout: 'pipe',
    stderr: 'pipe',
    reject: false,
  });
  proc.stdout?.on('data', (chunk: Buffer) => {
    opts.onChunk('stdout', chunk.toString('utf8'), seq++);
  });
  proc.stderr?.on('data', (chunk: Buffer) => {
    opts.onChunk('stderr', chunk.toString('utf8'), seq++);
  });
  void proc.then((result) => {
    opts.onExit(result.exitCode ?? 1, Date.now() - startTime);
  });
  return {
    runId,
    async cancel(): Promise<void> {
      // Reuse Windows taskkill + Unix SIGTERM/SIGKILL pattern from executor/single.ts killAndWait
      const IS_WINDOWS = process.platform === 'win32';
      if (IS_WINDOWS && proc.pid) {
        try { execSync(`taskkill /f /t /pid ${proc.pid}`, { stdio: 'pipe' }); } catch {}
      } else {
        proc.kill('SIGTERM');
      }
      const forceKillTimer = setTimeout(() => {
        if (IS_WINDOWS && proc.pid) {
          try { execSync(`taskkill /f /t /pid ${proc.pid}`, { stdio: 'pipe' }); } catch {}
        } else {
          proc.kill('SIGKILL');
        }
      }, 5_000);
      try { await proc; } catch {} finally { clearTimeout(forceKillTimer); }
    },
  };
}
```
