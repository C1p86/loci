// packages/server/src/services/dispatcher.ts
export interface QueueEntry {
  runId: string;
  orgId: string;
  taskSnapshot: TaskSnapshot;
  params: Record<string, string>;    // already resolved by dispatch-resolver (Phase 9)
  labelRequirements: string[];       // ["os=linux", "arch=x64"]
  timeoutSeconds: number;
}
export class DispatchQueue {
  private queue: QueueEntry[] = [];
  private timer: NodeJS.Timeout | null = null;
  private lastDispatchedAgentCursor = new Map<string, string>();
  enqueue(entry: QueueEntry): void { this.queue.push(entry); }
  dequeue(runId: string): void { this.queue = this.queue.filter((e) => e.runId !== runId); }
  countByOrg(orgId: string): number { return this.queue.filter((e) => e.orgId === orgId).length; }
  start(tickFn: () => Promise<void>, intervalMs = 250): void {
    if (this.timer) return;
    this.timer = setInterval(() => { void tickFn(); }, intervalMs);
    this.timer.unref();
  }
  stop(): void {
    if (this.timer) { clearInterval(this.timer); this.timer = null; }
  }
}

// packages/server/src/services/agent-selector.ts
import { and, eq, inArray, sql } from 'drizzle-orm';
import { agents, taskRuns } from '../db/schema.js';

export async function selectEligibleAgent(
  db: PostgresJsDatabase,
  orgId: string,
  labelRequirements: string[],
  lastCursorAgentId: string | null,  // for round-robin tiebreak
): Promise<string | null> {
  // Build JSONB requirement object
  const reqObject: Record<string, string> = {};
  for (const req of labelRequirements) {
    const idx = req.indexOf('=');
    if (idx === -1) continue;
    reqObject[req.slice(0, idx)] = req.slice(idx + 1);
  }
  const labelJson = JSON.stringify(reqObject);

  // Subquery: count active runs per agent
  const activeRunsSq = db
    .select({
      agentId: taskRuns.agentId,
      cnt: sql<number>`count(*)::int`.as('cnt'),
    })
    .from(taskRuns)
    .where(inArray(taskRuns.state, ['dispatched', 'running']))
    .groupBy(taskRuns.agentId)
    .as('active_runs');

  const candidates = await db
    .select({
      agentId: agents.id,
      maxConcurrent: agents.maxConcurrent,
      activeCount: sql<number>`coalesce(${activeRunsSq.cnt}, 0)`.as('active_count'),
    })
    .from(agents)
    .leftJoin(activeRunsSq, eq(agents.id, activeRunsSq.agentId))
    .where(
      and(
        eq(agents.orgId, orgId),
        eq(agents.state, 'online'),
        sql`${agents.lastSeenAt} > now() - interval '60 seconds'`,
        sql`${agents.labels} @> ${labelJson}::jsonb`,
        sql`coalesce(${activeRunsSq.cnt}, 0) < ${agents.maxConcurrent}`,
      ),
    )
    .orderBy(sql`coalesce(${activeRunsSq.cnt}, 0) ASC`, agents.id);  // least-busy, then id for deterministic tiebreak

  if (candidates.length === 0) return null;

  // Round-robin tiebreak: if multiple candidates tied on activeCount, rotate past lastCursor
  const minCount = candidates[0]!.activeCount;
  const tied = candidates.filter((c) => c.activeCount === minCount);
  if (tied.length === 1) return tied[0]!.agentId;
  // Find first tied agent AFTER lastCursorAgentId in the sorted list; wrap around
  if (lastCursorAgentId) {
    const idx = tied.findIndex((c) => c.agentId === lastCursorAgentId);
    if (idx !== -1 && idx + 1 < tied.length) return tied[idx + 1]!.agentId;
  }
  return tied[0]!.agentId;
}

// packages/server/src/services/timeout-manager.ts
const runTimers = new Map<string, NodeJS.Timeout>();

export function registerRunTimer(fastify, runId, timeoutSeconds) {
  const existing = runTimers.get(runId);
  if (existing) clearTimeout(existing);
  const cappedSeconds = Math.min(timeoutSeconds, 86400);   // D-20 server-side max 24h
  const timer = setTimeout(
    async () => { runTimers.delete(runId); await handleRunTimeout(fastify, runId); },
    cappedSeconds * 1000,
  );
  timer.unref();
  runTimers.set(runId, timer);
}

export function cancelRunTimer(runId: string): void {
  const timer = runTimers.get(runId);
  if (timer) { clearTimeout(timer); runTimers.delete(runId); }
}

export function clearAllRunTimers(): void {
  for (const t of runTimers.values()) clearTimeout(t);
  runTimers.clear();
}

async function handleRunTimeout(fastify, runId): Promise<void> {
  const repos = makeRepos(fastify.db, fastify.mek);
  // We need orgId → but handleRunTimeout only has runId. Solution: SELECT org_id first (admin cross-org lookup).
  const run = await fastify.db.select().from(taskRuns).where(eq(taskRuns.id, runId)).limit(1);
  const row = run[0];
  if (!row) return;
  const updated = await repos.forOrg(row.orgId).taskRuns.updateStateMulti(
    runId,
    ['dispatched', 'running'],
    'timed_out',
    { finishedAt: sql`now()`, exitCode: -1 },
  );
  if (!updated) return;  // already terminal
  // Send cancel frame to agent if still connected
  if (updated.agentId) {
    const ws = fastify.agentRegistry.get(updated.agentId);
    if (ws?.readyState === WS.OPEN) {
      ws.send(JSON.stringify({ type: 'cancel', run_id: runId, reason: 'timeout' }));
    }
  }
  fastify.log.warn({ runId }, 'run timed out');
}

// packages/server/src/services/reconciler.ts
export async function runBootReconciliation(fastify: FastifyInstance): Promise<void> {
  const runs = await makeRepos(fastify.db, fastify.mek).admin.findRunsForReconciliation();
  for (const run of runs) {
    // Step A: queued runs
    if (run.state === 'queued') {
      fastify.dispatchQueue.enqueue(toQueueEntry(run));
      continue;
    }
    // Step B: dispatched/running — check timeout window first
    const dispatchedMs = run.dispatchedAt ? new Date(run.dispatchedAt).getTime() : Date.now();
    const expiresAt = dispatchedMs + run.timeoutSeconds * 1000;
    if (Date.now() > expiresAt) {
      await makeRepos(fastify.db, fastify.mek).forOrg(run.orgId).taskRuns.updateStateMulti(
        run.id, ['dispatched', 'running'], 'timed_out',
        { finishedAt: sql`now()`, exitCode: -1 },
      );
      fastify.log.warn({ runId: run.id }, 'reconciliation: timeout expired during downtime');
      continue;
    }
    // Step C: agent connected check
    const agentConnected = run.agentId && fastify.agentRegistry.has(run.agentId);
    if (agentConnected) {
      const remaining = Math.max(1, run.timeoutSeconds - Math.floor((Date.now() - dispatchedMs) / 1000));
      registerRunTimer(fastify, run.id, remaining);
      continue;
    }
    // Step D: agent gone — D-23 Pitfall 6 rule: only re-queue dispatched; orphan running
    if (run.state === 'dispatched') {
      await forOrg(run.orgId).taskRuns.updateState(run.id, 'dispatched', 'queued', { agentId: null });
      fastify.dispatchQueue.enqueue(toQueueEntry(run));
      fastify.log.info({ runId: run.id }, 'reconciliation: re-queued (agent gone)');
    } else {
      await forOrg(run.orgId).taskRuns.updateState(run.id, 'running', 'orphaned',
        { finishedAt: sql`now()`, exitCode: -1 });
      fastify.log.warn({ runId: run.id }, 'reconciliation: orphaned (agent gone, had side effects)');
    }
  }
}

// Helper for reconnect reconciliation — activates Phase 8 D-18 stub
export async function buildReconnectReconciliation(
  fastify: FastifyInstance,
  orgId: string,
  runningRuns: RunState[],
): Promise<ReconcileEntry[] > {
  const result: ReconcileEntry[] = [];
  const repo = makeRepos(fastify.db, fastify.mek).forOrg(orgId).taskRuns;
  for (const agentRun of runningRuns) {
    const dbRun = await repo.getById(agentRun.run_id);
    if (!dbRun) { result.push({ run_id: agentRun.run_id, action: 'abandon' }); continue; }
    if (['succeeded','failed','cancelled','timed_out','orphaned'].includes(dbRun.state)) {
      result.push({ run_id: agentRun.run_id, action: 'abandon' });
      continue;
    }
    // Active — promote dispatched→running if needed; register remaining timer
    if (dbRun.state === 'dispatched') {
      await repo.updateState(agentRun.run_id, 'dispatched', 'running', { startedAt: sql`now()` });
    }
    const dispatchedAt = dbRun.dispatchedAt ? new Date(dbRun.dispatchedAt).getTime() : Date.now();
    const elapsed = Math.floor((Date.now() - dispatchedAt) / 1000);
    const remaining = Math.max(1, dbRun.timeoutSeconds - elapsed);
    registerRunTimer(fastify, agentRun.run_id, remaining);
    result.push({ run_id: agentRun.run_id, action: 'continue' });
  }
  return result;
}

let ticking = false;
async function tickDispatcher(fastify: FastifyInstance): Promise<void> {
  if (ticking) return;          // Pitfall 2 — concurrent tick guard
  ticking = true;
  try {
    const queue = fastify.dispatchQueue;
    const snapshot = queue.getEntries();  // new method: returns read-only copy
    for (const entry of snapshot) {
      const agentId = await selectEligibleAgent(
        fastify.db, entry.orgId, entry.labelRequirements,
        queue.getLastCursor(entry.orgId),
      );
      if (!agentId) continue;  // no eligible agent → leave in queue for next tick
      // CAS queued → dispatched
      const updated = await makeRepos(fastify.db, fastify.mek).forOrg(entry.orgId).taskRuns.updateState(
        entry.runId, 'queued', 'dispatched',
        { agentId, dispatchedAt: sql`now()` },
      );
      if (!updated) { queue.dequeue(entry.runId); continue; }  // won by someone else — drop
      queue.dequeue(entry.runId);
      queue.setLastCursor(entry.orgId, agentId);
      // Send dispatch frame
      const ws = fastify.agentRegistry.get(agentId);
      if (ws?.readyState === WS.OPEN) {
        ws.send(JSON.stringify({
          type: 'dispatch', run_id: entry.runId, task_snapshot: entry.taskSnapshot,
          params: entry.params, timeout_seconds: entry.timeoutSeconds,
        }));
      }
      // Register timeout timer
      registerRunTimer(fastify, entry.runId, entry.timeoutSeconds);
    }
  } catch (err) {
    fastify.log.error({ err }, 'tickDispatcher error');
  } finally {
    ticking = false;
  }
}
