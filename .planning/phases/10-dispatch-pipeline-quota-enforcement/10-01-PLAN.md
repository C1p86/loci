async consumeRegistrationToken(tokenId: string): Promise<string> {
  const rows = await db
    .update(registrationTokens)
    .set({ consumedAt: sql`now()` })

// packages/server/src/repos/task-runs.ts — updateState helper
async function updateState(
  db: PostgresJsDatabase,
  orgId: string,
  runId: string,
  expectedState: TaskRunState,
  newState: TaskRunState,
  extra?: Partial<NewTaskRun>,
): Promise<TaskRun | undefined> {
  const rows = await db
    .update(taskRuns)
    .set({ state: newState, updatedAt: sql`now()`, ...extra })
    .where(and(
      eq(taskRuns.id, runId),
      eq(taskRuns.orgId, orgId),      // org scope guard (see Pitfall 5 — frame spoofing)
      eq(taskRuns.state, expectedState),
    ))
    .returning();
  return rows[0];
}
