WHY after consumeRegistrationToken: the token is single-use; consuming it BEFORE the quota check means a rejected 6th-agent attempt still burns the token. async function handleResultFrame(fastify, conn, frame) {
```typescript
  // Step 1: cross-tenant guard (Pitfall 5)
  const taskRunsRepo = makeRepos(fastify.db, fastify.mek).forOrg(conn.orgId).taskRuns;
  const belongs = await taskRunsRepo.verifyBelongsToOrg(frame.run_id);
  if (!belongs) {
    send(socket, { type: 'error', code: 'AUTHZ_RUN_CROSS_ORG', message: 'run not found', close: false });
    fastify.log.warn({ agentId: conn.agentId, runId: frame.run_id, orgId: conn.orgId }, 'frame spoofing attempt');
    return;
  }
  // Step 2: cancel run timer BEFORE DB write (Pitfall 1 discipline)
  //   — imported from services/timeout-manager.ts (created in Plan 10-03).
  //   — In Plan 10-02: import as `import { cancelRunTimer } from '../services/timeout-manager.js'`
  //     timeout-manager.ts does NOT exist yet until Plan 10-03. Solution: add a minimal stub in
  //     services/timeout-manager.ts in this plan that exports empty functions; Plan 10-03 replaces
  //     with full implementation. This lets handler.ts import cleanly in Plan 10-02 without a
  //     phantom symbol.
  cancelRunTimer(frame.run_id);
  // Step 3: CAS transition
  const targetState = frame.cancelled ? 'cancelled' : (frame.exit_code === 0 ? 'succeeded' : 'failed');
  const updated = await taskRunsRepo.updateStateMulti(
    frame.run_id,
    ['running', 'dispatched'],
    targetState,
    { exitCode: frame.exit_code, finishedAt: sql`now()` },
  );
  // CAS loser (already terminal — e.g. timed_out arrived first): silent, debug log only
  if (!updated) fastify.log.debug({ runId: frame.run_id }, 'result frame for terminal run — ignored');
}
```
