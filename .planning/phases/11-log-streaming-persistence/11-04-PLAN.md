Step 1 — edit `packages/xci/src/agent/runner.ts`. Keep the existing Windows taskkill logic untouched. Append:

```typescript
export function redactLine(line: string, values: readonly string[] | undefined): string {
  if (!values || values.length === 0) return line;
  let out = line;
  for (const v of values) out = out.replaceAll(v, '***');
  return out;
}

const MAX_CHUNK_BYTES = 8192;

export function splitChunk(data: string, maxBytes: number): string[] {
  if (data.length === 0) return [];
  if (Buffer.byteLength(data, 'utf8') <= maxBytes) return [data];
  const pieces: string[] = [];
  let current = '';
  let currentBytes = 0;
  // Iterate code points (handles multi-byte correctly)
  for (const ch of data) {
    const chBytes = Buffer.byteLength(ch, 'utf8');
    if (currentBytes + chBytes > maxBytes && current.length > 0) {
      pieces.push(current);
      current = '';
      currentBytes = 0;
    }
    current += ch;
    currentBytes += chBytes;
  }
  if (current.length > 0) pieces.push(current);
  return pieces;
}
```

Agent log_chunk frames are persisted to Postgres (log_chunks table) and fanned out to subscribed UI clients.
