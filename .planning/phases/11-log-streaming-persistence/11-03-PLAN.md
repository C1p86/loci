Step 1 — create `packages/server/src/routes/runs/download.ts` mirroring trigger.ts's plugin shape. Key code:

```typescript
import type { FastifyPluginAsync } from 'fastify';
import { RunNotFoundError, SessionRequiredError } from '../../errors.js';
import { makeRepos } from '../../repos/index.js';
import { requireAnyMember } from './helpers.js';

export const downloadLogRoute: FastifyPluginAsync = async (fastify) => {
  fastify.get<{ Params: { orgId: string; runId: string } }>(
    '/:orgId/runs/:runId/logs.log',
    { preHandler: [fastify.requireAuth] },
    async (req, reply) => {
      requireAnyMember(req);
      const orgId = req.org?.id;
      if (!orgId) throw new SessionRequiredError();
      const { runId } = req.params;

      const repos = makeRepos(fastify.db, fastify.mek);
      const run = await repos.forOrg(orgId).taskRuns.getById(runId);
      if (!run) throw new RunNotFoundError();

      reply.header('Content-Type', 'text/plain; charset=utf-8');
      reply.header('Content-Disposition', `attachment; filename="run-${runId}.log"`);
      reply.header('Cache-Control', 'no-store');
      reply.hijack();

      const raw = reply.raw;
      const LIMIT = 1000;
      let lastSeq = -1;
      // Stream paged chunks — never buffer entire log (D-15)
      while (true) {
        const rows = await repos.forOrg(orgId).logChunks.getByRunId(runId, {
          sinceSeq: lastSeq, limit: LIMIT,
        });
        if (rows.length === 0) break;
        for (const row of rows) {
          const tsIso = row.ts.toISOString();
          const prefix = `[${tsIso} ${row.stream.toUpperCase()}] `;
          const suffix = row.data.endsWith('\n') ? '' : '\n';
          raw.write(`${prefix}${row.data}${suffix}`);
        }
        lastSeq = rows[rows.length - 1]!.seq;
        if (rows.length < LIMIT) break;
      }
      raw.end();
    },
  );
};
```

```typescript
import type { FastifyPluginAsync } from 'fastify';
import type { WebSocket } from 'ws';
import { LogSubscriptionUnauthorizedError, RunNotFoundError, SessionRequiredError } from '../../errors.js';
import { makeRepos } from '../../repos/index.js';
import { requireAnyMember } from './helpers.js';

interface SubscribeFrame { type: 'subscribe'; sinceSeq?: number; }

function parseSubscribeFrame(raw: string): SubscribeFrame | null {
  try {
    const o = JSON.parse(raw);
    if (typeof o !== 'object' || o === null) return null;
    if (o.type !== 'subscribe') return null;
    if (o.sinceSeq !== undefined && (typeof o.sinceSeq !== 'number' || !Number.isInteger(o.sinceSeq) || o.sinceSeq < -1)) return null;
    return { type: 'subscribe', sinceSeq: o.sinceSeq };
  } catch { return null; }
}

export const logsWsRoute: FastifyPluginAsync = async (fastify) => {
  fastify.get<{ Params: { orgId: string; runId: string } }>(
    '/ws/orgs/:orgId/runs/:runId/logs',
    { websocket: true, preHandler: [fastify.requireAuth] },
    async (socket, req) => {
      try {
        requireAnyMember(req);
      } catch (err) {
        socket.send(JSON.stringify({ type: 'error', code: (err as { code?: string }).code ?? 'AUTHZ_DENIED' }));
        socket.close(1008, 'unauthorized');
        return;
      }
      const orgId = req.org?.id;
      if (!orgId) { socket.close(1008, 'no session'); return; }
      const { runId } = req.params as { orgId: string; runId: string };

      const repos = makeRepos(fastify.db, fastify.mek);
      const run = await repos.forOrg(orgId).taskRuns.getById(runId);
      if (!run) {
        socket.send(JSON.stringify({ type: 'error', code: 'NF_RUN' }));
        socket.close(1008, 'not found');
        return;
      }

      // Await first frame — subscribe handshake
      let subscribed = false;
      socket.on('message', async (data: Buffer) => {
        if (subscribed) return; // additional frames after subscribe are ignored (future extension)
        const frame = parseSubscribeFrame(data.toString('utf8'));
        if (!frame) { socket.close(4002, 'frame_invalid'); return; }
        subscribed = true;

        const startFrom = frame.sinceSeq ?? -1;
        const LIMIT = 1000;
        let lastSeq = startFrom;
        while (true) {
          const rows = await repos.forOrg(orgId).logChunks.getByRunId(runId, { sinceSeq: lastSeq, limit: LIMIT });
          if (rows.length === 0) break;
          for (const row of rows) {
            if (socket.readyState !== 1) return;
            socket.send(JSON.stringify({ type: 'chunk', seq: row.seq, stream: row.stream, data: row.data, ts: row.ts.toISOString() }));
          }
          lastSeq = rows[rows.length - 1]!.seq;
          if (rows.length < LIMIT) break;
        }

        // Terminal state check — if run is already done, emit end frame + close
        const runAfter = await repos.forOrg(orgId).taskRuns.getById(runId);
        const terminal = runAfter && ['succeeded','failed','cancelled','timed_out','orphaned'].includes(runAfter.state);
        if (terminal) {
          socket.send(JSON.stringify({ type: 'end', state: runAfter!.state, exitCode: runAfter!.exitCode }));
          setTimeout(() => { try { socket.close(1000); } catch {} }, 5000).unref();
          return;
        }

        // Register as live subscriber (LogFanout wires its own ws.on('close') for removal)
        fastify.logFanout.addSubscriber(runId, orgId, socket as WebSocket);
      });

      socket.on('close', () => { /* LogFanout handles removal on its wired close listener */ });
    },
  );
};
```

```typescript
import type { FastifyInstance } from 'fastify';
import { LogRetentionJobError } from '../errors.js';
import { makeRepos } from '../repos/index.js';

export async function runRetentionCleanup(fastify: FastifyInstance): Promise<void> {
  try {
    const repos = makeRepos(fastify.db, fastify.mek);
    const result = await repos.admin.runRetentionCleanup({ batchSize: 10_000, maxIterations: 100 });
    fastify.log.info({
      rowsDeleted: result.rowsDeleted,
      iterations: result.iterations,
      orgs: Object.keys(result.perOrg).length,
      perOrg: result.perOrg,
    }, 'log retention cleanup complete');
  } catch (err) {
    if (err instanceof LogRetentionJobError) {
      fastify.log.error({ err }, 'log retention cleanup failed — will retry on next interval');
    } else {
      fastify.log.error({ err }, 'log retention cleanup unexpected error');
    }
    // D-17: best-effort; do NOT propagate so the setInterval keeps firing.
  }
}

export function startLogRetentionJob(fastify: FastifyInstance): void {
  // D-20: immediate boot pass so long downtimes don't leave retention unprocessed.
  void runRetentionCleanup(fastify);

  const intervalMs = fastify.config.LOG_RETENTION_INTERVAL_MS;
  const handle = setInterval(() => {
    void runRetentionCleanup(fastify);
  }, intervalMs);
  handle.unref();
  fastify.logRetentionTimer = handle;
}
```
