<!-- WS protocol (already shipped Phase 11) -->

URL: /ws/orgs/:orgId/runs/:runId/logs
Auth: cookie-based (xci_sid) — works because /ws is proxied by Vite dev server preserving cookies; in prod the server serves both WS and HTTP

Client → server frames:
- {type:'subscribe', sinceSeq: number}   // sent immediately after onopen

Server → client frames:
- {type:'chunk', seq:number, stream:'stdout'|'stderr', ts:string, data:string}
- {type:'gap', fromSeq:number, toSeq:number}          // dropped due to backpressure (LOG-07)
- {type:'end',  state: RunState, exitCode: number|null}
- {type:'catchup_complete'}                            // sent after sinceSeq backlog is drained

Close codes:
- 1000 normal
- 4001 unauthorized (shouldn't happen after cookie auth passed)
- 4004 run not found / not in org
- 4008 max subscribers exceeded

Download endpoint: GET /api/orgs/:orgId/runs/:runId/logs.log
- 200 text/plain attachment (Content-Disposition)
- Requires session cookie + org-scoped

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| server WS frame → LogViewer DOM | Must render chunk `data` as plain text (no HTML execution) |
| client tab inactive → WS stays open | Backgrounded tab still receives chunks but browsers may throttle; reconnect logic must handle resume |
| registration token / session cookie → WS auth | Cookie-auth on upgrade; token never in URL per ATOK-03 (N/A for log WS but principle applies) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-13-04-01 | X (XSS) | Log chunk `data` rendered in DOM | mitigate | Render exclusively as `{chunk.data}` in React (auto-escaped) inside a `<pre>` element. NEVER via dangerouslySetInnerHTML. Explicit test asserting XSS input `<script>alert(1)</script>` renders as literal text, not HTML. |
| T-13-04-02 | I (Info Disclosure) | Secret leak in log chunks | accept | Server already redacts org + agent secrets pre-persist (LOG-06, Phase 11 D-06). Web renders what the server sends; client-side redaction would be defense-in-depth but LOG-06 is authoritative. |
| T-13-04-03 | D (DoS) | Huge log runs crashing browser | mitigate | LogViewer caps retained chunks at 50,000; older chunks marked "truncated — download full log" with link to /logs.log. Virtualization keeps DOM nodes bounded (react-virtuoso OR hand-rolled windowing). |
| T-13-04-04 | D (DoS) | WS reconnect loop on persistent server failure | mitigate | Backoff caps at 30s; after 10 consecutive failures, `wsStore.status='disconnected'` stays red; retry button appears; user can navigate away to stop loop. |
| T-13-04-05 | T (Tampering) | Stale sinceSeq after network partition | mitigate | Client tracks `lastSeenSeq` from every received chunk frame. Reconnect subscribes with `sinceSeq: lastSeenSeq + 1`. Server guarantees ordered replay of persisted chunks (Phase 11 SC-1). |
| T-13-04-06 | R (Repudiation) | User disputes run state shown | mitigate | `{type:'end'}` frame is the authoritative source of run state change; UI updates from this frame. Fallback polling of /runs/:runId (5s) also returns authoritative state. |
| T-13-04-07 | I (Info Disclosure) | Tab-switch reveals half-redacted logs | accept | Server-side redaction is complete before write; browser-visible history is the same as server persistence. No additional risk. |

## Security hardening mandatory in this plan
- [x] Log chunks rendered via {text} in `<pre>` — NEVER dangerouslySetInnerHTML
- [x] Explicit XSS test: inject `<script>` in mock chunk; assert rendered as literal text
- [x] WS URL uses ws:// relative path through Vite proxy; no hardcoded cross-origin (production uses same-origin + wss)
- [x] sinceSeq arithmetic bounded (server clamps to persisted range; client doesn't send negative)
- [x] Memory cap: 50,000 chunks retained; older truncated with link
- [x] Download endpoint link rendered as `<a download>` (browser handles attachment); no fetch-to-blob in memory
</threat_model>

    ```ts
    import { useEffect, useRef, useState } from 'react';
    import type { LogChunk, RunState } from '../lib/types.js';
    import { useWsStore } from '../stores/wsStore.js';

    const MAX_CHUNKS = 50_000;
    const BACKOFFS = [1000, 2000, 4000, 8000, 16000, 30000];

    interface Args { orgId: string | undefined; runId: string | undefined; enabled: boolean; }

    export function useLogWebSocket({ orgId, runId, enabled }: Args) {
      const [chunks, setChunks] = useState<LogChunk[]>([]);
      const [endState, setEndState] = useState<RunState | null>(null);
      const [exitCode, setExitCode] = useState<number | null>(null);
      const setStatus = useWsStore((s) => s.setStatus);

      const lastSeenSeqRef = useRef(-1);
      const backoffIdxRef = useRef(0);
      const wsRef = useRef<WebSocket | null>(null);
      const reconnectTimerRef = useRef<number | null>(null);
      const mountedRef = useRef(true);

      useEffect(() => {
        mountedRef.current = true;
        if (!enabled || !orgId || !runId) { setStatus('disconnected'); return; }

        function connect() {
          setStatus(lastSeenSeqRef.current >= 0 ? 'reconnecting' : 'connected'); // first connect shows green quickly
          const proto = window.location.protocol === 'https:' ? 'wss:' : 'ws:';
          const url = `${proto}//${window.location.host}/ws/orgs/${orgId}/runs/${runId}/logs`;
          const ws = new WebSocket(url);
          wsRef.current = ws;

          ws.addEventListener('open', () => {
            if (!mountedRef.current) return;
            backoffIdxRef.current = 0;
            setStatus('connected');
            ws.send(JSON.stringify({ type: 'subscribe', sinceSeq: lastSeenSeqRef.current + 1 }));
          });

          ws.addEventListener('message', (ev) => {
            if (!mountedRef.current) return;
            let msg: unknown;
            try { msg = JSON.parse(String(ev.data)); } catch { return; }
            if (typeof msg !== 'object' || msg === null || !('type' in msg)) return;
            const m = msg as { type: string; [k: string]: unknown };
            if (m.type === 'chunk') {
              const chunk = m as unknown as LogChunk;
              lastSeenSeqRef.current = Math.max(lastSeenSeqRef.current, chunk.seq);
              setChunks((prev) => {
                const next = prev.length >= MAX_CHUNKS ? prev.slice(prev.length - MAX_CHUNKS + 1) : prev;
                return [...next, chunk];
              });
            } else if (m.type === 'gap') {
              // Synthesize a visual marker in-band — LogViewer renders it distinctly
              const markerSeq = Number(m.toSeq);
              lastSeenSeqRef.current = Math.max(lastSeenSeqRef.current, markerSeq);
              setChunks((prev) => [...prev, {
                seq: markerSeq, stream: 'stderr', ts: new Date().toISOString(),
                data: `[gap: ${m.fromSeq}–${m.toSeq} missed due to backpressure]`,
              }]);
            } else if (m.type === 'end') {
              setEndState(m.state as RunState);
              setExitCode(typeof m.exitCode === 'number' ? m.exitCode : null);
            } else if (m.type === 'catchup_complete') {
              // no-op — state is implicit
            }
          });

          ws.addEventListener('close', (ev) => {
            if (!mountedRef.current) return;
            wsRef.current = null;
            // Terminal close codes: do not reconnect
            if (ev.code === 1000 || ev.code === 4004 || ev.code === 4008) {
              setStatus('disconnected');
              return;
            }
            setStatus('reconnecting');
            const delay = BACKOFFS[Math.min(backoffIdxRef.current, BACKOFFS.length - 1)];
            backoffIdxRef.current += 1;
            reconnectTimerRef.current = window.setTimeout(() => {
              if (mountedRef.current) connect();
            }, delay);
          });

          ws.addEventListener('error', () => { /* close will fire next */ });
        }

        connect();

        return () => {
          mountedRef.current = false;
          if (reconnectTimerRef.current != null) window.clearTimeout(reconnectTimerRef.current);
          if (wsRef.current && wsRef.current.readyState <= WebSocket.OPEN) wsRef.current.close(1000, 'unmount');
          setStatus('disconnected');
        };
        // Important: re-run on runId / orgId / enabled changes only
      }, [orgId, runId, enabled, setStatus]);

      return { chunks, endState, exitCode };
    }
    ```

    ```tsx
    import { useEffect, useRef, useState } from 'react';
    import { useLogWebSocket } from '../hooks/useLogWebSocket.js';
    import { useUiStore } from '../stores/uiStore.js';
    import { useAuthStore } from '../stores/authStore.js';
    import { Button } from './ui/button.js';
    import { Download, Clock, ArrowDown } from 'lucide-react';
    import type { RunState } from '../lib/types.js';

    interface LogViewerProps { runId: string; initialState: RunState; }

    export function LogViewer({ runId, initialState }: LogViewerProps) {
      const orgId = useAuthStore((s) => s.org?.id);
      const timestampVisible = useUiStore((s) => s.logTimestampVisible);
      const setTimestampVisible = useUiStore((s) => s.setLogTimestampVisible);
      const autoscrollPaused = useUiStore((s) => s.logAutoscrollPaused);
      const setAutoscrollPaused = useUiStore((s) => s.setLogAutoscrollPaused);

      const isTerminal = (s: RunState) => ['succeeded','failed','cancelled','timed_out','orphaned'].includes(s);
      const enabled = !isTerminal(initialState);

      const { chunks, endState, exitCode } = useLogWebSocket({ orgId, runId, enabled });

      const containerRef = useRef<HTMLDivElement | null>(null);
      const bottomRef = useRef<HTMLDivElement | null>(null);
      const [newSinceScroll, setNewSinceScroll] = useState(0);

      // IntersectionObserver: autoscroll pause/resume (SC-3)
      useEffect(() => {
        if (!bottomRef.current) return;
        const observer = new IntersectionObserver(
          (entries) => {
            const visible = entries[0]?.isIntersecting ?? true;
            setAutoscrollPaused(!visible);
            if (visible) setNewSinceScroll(0);
          },
          { root: containerRef.current, rootMargin: '0px', threshold: 0.01 },
        );
        observer.observe(bottomRef.current);
        return () => observer.disconnect();
      }, [setAutoscrollPaused]);

      // Auto-scroll on new chunks when not paused
      useEffect(() => {
        if (!autoscrollPaused && bottomRef.current) {
          bottomRef.current.scrollIntoView({ behavior: 'auto', block: 'end' });
        } else if (autoscrollPaused) {
          setNewSinceScroll((n) => n + 1);
        }
      }, [chunks.length, autoscrollPaused]);

      function resumeAutoscroll() {
        bottomRef.current?.scrollIntoView({ behavior: 'smooth', block: 'end' });
        setAutoscrollPaused(false);
        setNewSinceScroll(0);
      }

      return (
        <div className="flex flex-col border rounded-md bg-black text-slate-100">
          <div className="flex items-center justify-between p-2 border-b border-white/10">
            <div className="text-sm">Logs — {chunks.length} chunk(s)</div>
            <div className="flex gap-2">
              <Button size="sm" variant="ghost" onClick={() => setTimestampVisible(!timestampVisible)}
                      title="Toggle timestamps" className="text-slate-200">
                <Clock className="h-4 w-4 mr-1" /> {timestampVisible ? 'Hide ts' : 'Show ts'}
              </Button>
              <a href={`/api/orgs/${orgId}/runs/${runId}/logs.log`} download target="_blank" rel="noopener noreferrer">
                <Button size="sm" variant="ghost" className="text-slate-200">
                  <Download className="h-4 w-4 mr-1" /> Download raw
                </Button>
              </a>
            </div>
          </div>
          {autoscrollPaused && (
            <button onClick={resumeAutoscroll}
              className="py-1 text-sm bg-amber-500/20 border-y border-amber-500/40 text-amber-200 hover:bg-amber-500/30">
              Autoscroll paused — {newSinceScroll} new line(s). Click to resume.
            </button>
          )}
          <div ref={containerRef} className="overflow-auto max-h-[65vh] min-h-[300px] p-2 font-mono text-xs" role="log">
            <pre className="whitespace-pre-wrap break-words">
              {chunks.map((c) => (
                <span key={c.seq} className={c.stream === 'stderr' ? 'text-red-300' : ''}>
                  {timestampVisible && <span className="text-slate-500">[{c.ts}] </span>}
                  {c.data}
                  {!c.data.endsWith('\n') ? '\n' : ''}
                </span>
              ))}
            </pre>
            <div ref={bottomRef} style={{ height: 1 }} />
          </div>
          {endState && (
            <div className="p-2 border-t border-white/10 text-sm">
              Run finished: <span className="font-semibold">{endState}</span>
              {exitCode != null && <span className="ml-2 text-slate-400">(exit {exitCode})</span>}
              <Button size="sm" variant="ghost" className="ml-3 text-slate-200" onClick={resumeAutoscroll}>
                <ArrowDown className="h-4 w-4" /> Jump to end
              </Button>
            </div>
          )}
        </div>
      );
    }
    ```
    Note: chunk data rendered via `{c.data}` text — auto-escaped by React. The `<pre whitespace-pre-wrap>` preserves formatting without allowing HTML.

    ```ts
    import { apiGet } from '../lib/api.js';
    import { useAuthStore } from '../stores/authStore.js';
    import type { RunSummary, RunState } from '../lib/types.js';

    export interface HistoryFilters {
      states?: RunState[]; taskId?: string; from?: string; to?: string;
      cursor?: string; limit?: number;
    }
    interface HistoryResponse { ok: true; runs: RunSummary[]; nextCursor?: string; }

    export function useRunHistory(filters: HistoryFilters) {
      const orgId = useAuthStore((s) => s.org?.id);
      const qs = new URLSearchParams();
      if (filters.states?.length) qs.set('state', filters.states.join(','));
      if (filters.taskId) qs.set('taskId', filters.taskId);
      if (filters.from)   qs.set('from', filters.from);
      if (filters.to)     qs.set('to', filters.to);
      if (filters.cursor) qs.set('cursor', filters.cursor);
      qs.set('limit', String(filters.limit ?? 25));
      return useQuery({
        queryKey: ['runs', 'history', orgId, Object.fromEntries(qs)],
        queryFn: () => apiGet<HistoryResponse>(`/api/orgs/${orgId}/runs?${qs.toString()}`),
        enabled: !!orgId,
      });
    }
    ```

    ```tsx
    import { Link } from 'react-router-dom';
    import { useState } from 'react';
    import { useRunHistory, type HistoryFilters } from '../../hooks/useRunHistory.js';
    import { useTasks } from '../../hooks/useTasks.js';
    import { Table, TableBody, TableCell, TableHead, TableHeader, TableRow } from '../../components/ui/table.js';
    import { Button } from '../../components/ui/button.js';
    import { Input } from '../../components/ui/input.js';
    import { Label } from '../../components/ui/label.js';
    import type { RunState } from '../../lib/types.js';

    const ALL_STATES: RunState[] = ['queued','dispatched','running','succeeded','failed','cancelled','timed_out','orphaned'];

    export function HistoryList() {
      const [filters, setFilters] = useState<HistoryFilters>({ limit: 25 });
      const [cursorStack, setCursorStack] = useState<string[]>([]);
      const { data: tasksData } = useTasks();
      const { data, isLoading, error } = useRunHistory(filters);

      function toggleState(s: RunState) {
        setFilters((f) => {
          const current = new Set(f.states ?? []);
          if (current.has(s)) current.delete(s); else current.add(s);
          return { ...f, states: Array.from(current), cursor: undefined };
        });
        setCursorStack([]);
      }
      function next() {
        if (data?.nextCursor) {
          setCursorStack((s) => [...s, filters.cursor ?? '']);
          setFilters((f) => ({ ...f, cursor: data.nextCursor }));
        }
      }
      function prev() {
        setCursorStack((s) => {
          const last = s[s.length - 1];
          setFilters((f) => ({ ...f, cursor: last || undefined }));
          return s.slice(0, -1);
        });
      }

      return (
        <div>
          <h1 className="text-2xl font-semibold mb-4">History</h1>
          <div className="mb-4 space-y-3 p-3 border rounded bg-card">
            <div className="flex flex-wrap gap-3">
              {ALL_STATES.map((s) => (
                <label key={s} className="flex items-center gap-1 text-sm">
                  <input type="checkbox" checked={(filters.states ?? []).includes(s)} onChange={() => toggleState(s)} />
                  {s}
                </label>
              ))}
            </div>
            <div className="flex gap-3 items-end">
              <div>
                <Label htmlFor="task-filter">Task</Label>
                <select id="task-filter" className="border rounded px-2 py-1"
                        value={filters.taskId ?? ''}
                        onChange={(e) => setFilters((f) => ({ ...f, taskId: e.target.value || undefined, cursor: undefined }))}>
                  <option value="">All tasks</option>
                  {(tasksData ?? []).map((t) => <option key={t.id} value={t.id}>{t.name}</option>)}
                </select>
              </div>
              <div>
                <Label htmlFor="from">From</Label>
                <Input id="from" type="datetime-local" onChange={(e) => setFilters((f) => ({ ...f, from: e.target.value || undefined, cursor: undefined }))} />
              </div>
              <div>
                <Label htmlFor="to">To</Label>
                <Input id="to" type="datetime-local" onChange={(e) => setFilters((f) => ({ ...f, to: e.target.value || undefined, cursor: undefined }))} />
              </div>
            </div>
          </div>
          {isLoading && <div>Loading…</div>}
          {error && <div className="text-destructive">{(error as Error).message}</div>}
          {data && (
            <>
              <Table>
                <TableHeader><TableRow>
                  <TableHead>Run</TableHead><TableHead>Task</TableHead><TableHead>State</TableHead><TableHead>Exit</TableHead><TableHead>Started</TableHead><TableHead>Finished</TableHead><TableHead>Trigger</TableHead>
                </TableRow></TableHeader>
                <TableBody>
                  {data.runs.map((r) => (
                    <TableRow key={r.id}>
                      <TableCell><Link to={`/runs/${r.id}`} className="hover:underline">{r.id.slice(-8)}</Link></TableCell>
                      <TableCell>{(tasksData ?? []).find((t) => t.id === r.taskId)?.name ?? r.taskId.slice(-8)}</TableCell>
                      <TableCell>{r.state}</TableCell>
                      <TableCell>{r.exitCode ?? '—'}</TableCell>
                      <TableCell>{r.startedAt ? new Date(r.startedAt).toLocaleString() : '—'}</TableCell>
                      <TableCell>{r.finishedAt ? new Date(r.finishedAt).toLocaleString() : '—'}</TableCell>
                      <TableCell>{r.triggerSource}</TableCell>
                    </TableRow>
                  ))}
                </TableBody>
              </Table>
              <div className="mt-3 flex gap-2">
                <Button variant="outline" disabled={cursorStack.length === 0} onClick={prev}>Previous</Button>
                <Button variant="outline" disabled={!data.nextCursor} onClick={next}>Next</Button>
              </div>
            </>
          )}
        </div>
      );
    }
    ```
