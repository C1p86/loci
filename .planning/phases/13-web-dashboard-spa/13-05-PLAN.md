<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| webhook token plaintext → one-time display | Shown once from POST response; never persisted in client state beyond component lifetime |
| GitHub webhook secret → server | Paste-once field; sent in POST body over TLS; server encrypts with org DEK |
| DLQ scrubbed payload → rendered JSON | Already scrubbed server-side per Phase 12 PLUG-08 — safe to render |
| Role change / member removal → server | Owner-only mutations; server re-checks role (Phase 7 D-15) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-13-05-01 | I (Info Disclosure) | Webhook token plaintext in DOM | mitigate | Plaintext token held ONLY in the mutation result state; never persisted to localStorage/sessionStorage; component unmounts after dismissal; tests assert no persistence. Same pattern as Plan 13-03 AgentsEmptyState (T-13-03-02). |
| T-13-05-02 | I (Info Disclosure) | GitHub webhook secret typed into form | mitigate | Input type=password + autocomplete=off; not logged to console; sent via apiPost over TLS only. Cleared on unmount via useEffect cleanup. |
| T-13-05-03 | X (XSS) | DLQ scrubbed payload rendered as JSON | mitigate | Stringify via JSON.stringify(..., null, 2) then render inside `<pre>` as text. React auto-escapes. NO dangerouslySetInnerHTML. |
| T-13-05-04 | E (Elevation) | Role change bypass | mitigate | RoleGate role="owner" wraps role change buttons. Server-side AUTH-10 enforcement remains authoritative. |
| T-13-05-05 | E (Elevation) | Leave org as last owner | mitigate | Leave-org button disabled if current user role === 'owner' (shown with tooltip "Transfer ownership before leaving"). Server enforces the same invariant (Phase 7 D-15: owner is non-removable). |
| T-13-05-06 | R (Repudiation) | DLQ retry audit | accept | Server records retriedAt + retryResult (Phase 12 schema); that's the audit. No additional client-side logging needed. |
| T-13-05-07 | I (Info Disclosure) | Webhook endpoint URL in plaintext token display | accept | Endpoint URL (e.g., `https://xci.example.com/hooks/github/:orgToken`) is necessarily public — the GitHub webhook config needs it. Token is the secret, not URL. |

## Security hardening mandatory in this plan
- [x] GitHub webhook secret input: type=password, autocomplete=off, cleared on unmount
- [x] Plaintext webhook token held in mutation.data state only; never persisted
- [x] DLQ payload rendered via `<pre>{JSON.stringify(...)}</pre>` (no innerHTML)
- [x] All mutations wrapped in RoleGate (invite, revoke, role change, leave org, create token, retry DLQ)
- [x] Leave-org button disabled for Owner + tooltip explanation

    ```ts
    import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
    import { apiGet, apiPost } from '../lib/api.js';
    import { useAuthStore } from '../stores/authStore.js';

    interface WebhookTokenRow { id: string; pluginName: 'github'|'perforce'; hasPluginSecret: boolean; createdAt: string; revokedAt: string|null; }
    interface CreateTokenResponse { ok: true; tokenId: string; plaintextToken: string; endpointUrl: string; }

    export function useWebhookTokens() {
      const orgId = useAuthStore((s) => s.org?.id);
      return useQuery({
        queryKey: ['webhookTokens', orgId],
        queryFn: () => apiGet<{ok:true; tokens: WebhookTokenRow[]}>(`/api/orgs/${orgId}/webhook-tokens`).then(r => r.tokens),
        enabled: !!orgId,
      });
    }
    export function useCreateWebhookToken() {
      const qc = useQueryClient();
      const orgId = useAuthStore((s) => s.org?.id);
      return useMutation({
        mutationFn: (args: { pluginName: 'github'|'perforce'; pluginSecret?: string }) =>
          apiPost<CreateTokenResponse>(`/api/orgs/${orgId}/webhook-tokens`, args),
        onSuccess: () => qc.invalidateQueries({ queryKey: ['webhookTokens', orgId] }),
      });
    }
    export function useRevokeWebhookToken() {
      const qc = useQueryClient();
      const orgId = useAuthStore((s) => s.org?.id);
      return useMutation({
        mutationFn: (args: { id: string }) => apiPost(`/api/orgs/${orgId}/webhook-tokens/${args.id}/revoke`),
        onSuccess: () => qc.invalidateQueries({ queryKey: ['webhookTokens', orgId] }),
      });
    }
    export type { WebhookTokenRow, CreateTokenResponse };
    ```

    ```ts
    export function useDlq(cursor?: string) {
      const orgId = useAuthStore((s) => s.org?.id);
      return useQuery({
        queryKey: ['dlq', orgId, cursor ?? 'first'],
        queryFn: () => apiGet<{ok:true; entries: DlqEntry[]; nextCursor?: string}>(`/api/orgs/${orgId}/dlq${cursor ? `?cursor=${encodeURIComponent(cursor)}` : ''}`),
        enabled: !!orgId,
      });
    }
    export function useDlqRetry() {
      const qc = useQueryClient();
      const orgId = useAuthStore((s) => s.org?.id);
      return useMutation({
        mutationFn: (args: { dlqId: string }) => apiPost<{ok:true; retryResult: DlqRetryResult}>(`/api/orgs/${orgId}/dlq/${args.dlqId}/retry`),
        onSuccess: () => qc.invalidateQueries({ queryKey: ['dlq', orgId] }),
      });
    }
    ```

    ```tsx
    function RetryBadge({ result }: { result: string|null }) {
      if (!result) return null;
      const colors: Record<string,string> = { succeeded: 'bg-green-100 text-green-800', failed_same_reason: 'bg-red-100 text-red-800', failed_new_reason: 'bg-amber-100 text-amber-900' };
      return <span className={`px-2 py-0.5 rounded text-xs ${colors[result] ?? ''}`}>{result}</span>;
    ```
