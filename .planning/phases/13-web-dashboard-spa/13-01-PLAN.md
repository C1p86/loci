    ```sql
    -- 1. Add tasks.slug + expose_badge with safe defaults
    ALTER TABLE "tasks" ADD COLUMN IF NOT EXISTS "slug" text NOT NULL DEFAULT '';
    ALTER TABLE "tasks" ADD COLUMN IF NOT EXISTS "expose_badge" boolean NOT NULL DEFAULT false;

    -- 2. Backfill tasks.slug from name (kebab-case, uniqueness ensured via CTE dedup)
    WITH slug_candidates AS (
      SELECT id, org_id,
        lower(regexp_replace(trim(name), '[^a-zA-Z0-9]+', '-', 'g')) AS base_slug,
        row_number() OVER (PARTITION BY org_id, lower(regexp_replace(trim(name), '[^a-zA-Z0-9]+', '-', 'g')) ORDER BY created_at) AS dup_rank
      FROM tasks
    )
    UPDATE tasks t
       SET slug = CASE
         WHEN sc.dup_rank = 1 THEN trim(both '-' from sc.base_slug)
         ELSE trim(both '-' from sc.base_slug) || '-' || substring(t.id from length(t.id)-5)
       END
      FROM slug_candidates sc
     WHERE sc.id = t.id;

    -- 3. Backfill orgs.slug if any row has empty slug (defensive — Phase 7 shipped orgs.slug NOT NULL)
    UPDATE orgs SET slug = lower(regexp_replace(trim(name), '[^a-zA-Z0-9]+', '-', 'g'))
     WHERE slug IS NULL OR slug = '';

    -- 4. Unique index on (org_id, slug) now that backfill is done
    CREATE UNIQUE INDEX IF NOT EXISTS "tasks_org_slug_unique" ON "tasks" ("org_id","slug");
    ```
    (If drizzle-kit already emitted `CREATE UNIQUE INDEX`, reorder statements so the UPDATE runs before the index.)

      const rows = await db.select({ id: orgs.id, name: orgs.name, slug: orgs.slug })

    import { type BadgeState, renderBadgeSvg } from './svg.js';

    export const registerBadgeRoutes: FastifyPluginAsync = async (fastify) => {
      fastify.get<{ Params: { orgSlug: string; taskSlug: string } }>(
        '/badge/:orgSlug/:taskSlug.svg',
        {
          config: {
            rateLimit: { max: 120, timeWindow: '1 minute' }, // D-32
          },
        },
        async (req, reply) => {
          const { orgSlug, taskSlug } = req.params;
          const { adminRepo } = req.repos;

          const state: BadgeState = await resolveBadgeState(adminRepo, orgSlug, taskSlug);
          const svg = renderBadgeSvg(state);

          reply
            .header('Content-Type', 'image/svg+xml; charset=utf-8')
            .header('Cache-Control', 'public, max-age=30')  // BADGE-02
            .header('X-Content-Type-Options', 'nosniff');
          return reply.send(svg);
        },
      );
    };

    async function resolveBadgeState(
      adminRepo: import('../../repos/index.js').AdminRepo,
      orgSlug: string,
      taskSlug: string,
    ): Promise<BadgeState> {
      const org = await adminRepo.findOrgBySlug(orgSlug);
      if (!org) return 'unknown';                          // BADGE-03
      const task = await adminRepo.findTaskByOrgAndSlug(org.id, taskSlug);
      if (!task || !task.exposeBadge) return 'unknown';    // BADGE-03, BADGE-04
      const run = await adminRepo.findLastTerminalRun(task.id);
      if (!run) return 'unknown';                          // no terminal run yet
      return run.state === 'succeeded' ? 'passing' : 'failing';
    }
