      // D-04 / PKG-05: serve @xci/web static bundle when WEB_STATIC_ROOT is configured.
      // Registered LAST so /api/*, /ws/*, /hooks/*, /badge/* all take precedence.
      // wildcard:false means static plugin does NOT catch unknown paths; we add an explicit
      // SPA fallback via setNotFoundHandler constrained to GET + non-API prefixes.
      if (app.config.WEB_STATIC_ROOT) {
        const { default: fastifyStatic } = await import('@fastify/static');
        await app.register(fastifyStatic, {
          root: app.config.WEB_STATIC_ROOT,
          prefix: '/',
          wildcard: false,
          decorateReply: false,
        });
        // SPA fallback: any GET request that did not match a route and does not start with
        // /api, /ws, /hooks, /badge should serve index.html so client-side routing works.
        app.setNotFoundHandler((req, reply) => {
          const url = req.raw.url ?? '/';
          const isApi = url.startsWith('/api/') || url.startsWith('/api?') || url === '/api';
          const isWs = url.startsWith('/ws/');
          const isHooks = url.startsWith('/hooks/');
          const isBadge = url.startsWith('/badge/');
          if (req.method !== 'GET' || isApi || isWs || isHooks || isBadge) {
            return reply.code(404).send({ error: 'NotFound' });
          }
          return reply.type('text/html').sendFile('index.html');
        });
      }
      ```

    # syntax=docker/dockerfile:1.7

    # =============================================================================
    # Stage 1: BUILDER — full toolchain, installs deps, runs turbo build, prunes via pnpm deploy.
    # node:22 (not slim) so we have build tools for any prebuilt-compile edge case.
    # =============================================================================
    FROM node:22 AS builder

    # Enable corepack + pin pnpm from root package.json packageManager field (D-15 reproducibility)
    RUN corepack enable && corepack prepare pnpm@10.33.0 --activate

    WORKDIR /workspace

    # Copy lockfile + workspace config FIRST for Docker layer caching (D-14).
    # When source changes but deps don't, the install layer is reused.
    COPY pnpm-workspace.yaml pnpm-lock.yaml package.json turbo.json ./
    COPY .changeset ./.changeset/
    COPY packages/xci/package.json ./packages/xci/
    COPY packages/server/package.json ./packages/server/
    COPY packages/web/package.json ./packages/web/

    # Install full workspace deps (devDeps included — needed for build)
    RUN pnpm install --frozen-lockfile

    # Now copy source (invalidates later layers, not the install layer above)
    COPY packages/xci ./packages/xci
    COPY packages/server ./packages/server
    COPY packages/web ./packages/web

    # Build both @xci/server and @xci/web (turbo handles dependency order)
    RUN pnpm turbo run build --filter=@xci/server --filter=@xci/web

    # D-03 / D-28: pnpm deploy produces a self-contained node_modules with PROD deps only
    # for the server package — no dev deps (no drizzle-kit, no tsx, no testcontainers).
    # --legacy flag writes a flat node_modules (no symlinks) that Docker COPY handles cleanly.
    RUN pnpm --filter=@xci/server deploy --prod --legacy /deploy

    # =============================================================================
    # Stage 2: RUNTIME — node:22-slim (glibc; required for @node-rs/argon2 prebuilt) (D-01/D-23)
    # =============================================================================
    FROM node:22-slim AS runtime

    # Install curl for HEALTHCHECK (minimal; slim image lacks it by default).
    # Pin via apt-get -y --no-install-recommends to keep layer small.
    RUN apt-get update \
     && apt-get install -y --no-install-recommends curl \
     && rm -rf /var/lib/apt/lists/*

    # D-25: non-root user UID/GID 10001 (uncommon, avoids collisions with host users).
    # -M skips creating a home dir; --system uses a low-range UID by default so we pass --uid.
    RUN groupadd --system --gid 10001 xci \
     && useradd --system --uid 10001 --gid 10001 --no-create-home --shell /usr/sbin/nologin xci

    WORKDIR /app

    # Copy pruned node_modules + built dist + migrations + web bundle from builder.
    # --chown flag sets ownership atomically during COPY (no chmod layer).
    COPY --from=builder --chown=xci:xci /deploy/node_modules ./node_modules
    COPY --from=builder --chown=xci:xci /deploy/package.json ./package.json
    COPY --from=builder --chown=xci:xci /workspace/packages/server/dist ./dist
    COPY --from=builder --chown=xci:xci /workspace/packages/server/drizzle ./drizzle
    COPY --from=builder --chown=xci:xci /workspace/packages/web/dist ./web

    # Fixed env pointing @fastify/static at the bundled web bundle.
    # Operator can override via `-e WEB_STATIC_ROOT=''` to disable; default ON in image.
    ENV NODE_ENV=production \
        PORT=3000 \
        WEB_STATIC_ROOT=/app/web

    USER 10001:10001

    EXPOSE 3000

    # D-01 HEALTHCHECK — NOTE: endpoint is /api/healthz (NOT /healthz).
    # start-period=20s covers boot-time migrations; interval=15s, retries=3.
    HEALTHCHECK --interval=15s --timeout=5s --retries=3 --start-period=30s \

    # process.on handlers. Exec form (JSON array) prevents shell wrapper from swallowing signals.
    CMD ["node", "dist/server.js"]
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Host → Container | Docker runtime isolates process; container user is UID 10001 (non-root) to limit damage from a container-escape |
| Container → Postgres | TCP connection over docker network; DATABASE_URL MAY contain credentials (not hardcoded; injected via env at runtime) |
| Client → Server (HTTP) | `/api/*` routes already mitigated in Phase 7 (auth, CSRF, rate-limit, helmet); @fastify/static adds a new attack surface (path traversal, MIME sniffing) |
| Build time → Image | Dockerfile COPY steps never touch .env / secrets files (enforced by .dockerignore) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-14-01-01 | Elevation of Privilege | Docker container runs as root | mitigate | USER 10001:10001 directive in Dockerfile; file ownership via --chown=xci:xci at COPY time. Verified by `docker run --rm xci/server:local id -u` returning 10001. |
| T-14-01-02 | Information Disclosure | Secrets baked into image | mitigate | .dockerignore excludes `.env`, `.env.*` (allowlist `.env.example` only). All runtime config via `-e` / `--env-file`. envSchema.required includes SESSION_COOKIE_SECRET + XCI_MASTER_KEY; server fails fast on missing. |
| T-14-01-03 | Tampering | drizzle-kit binary in runtime image could be used to run arbitrary migrations | mitigate | `pnpm --filter @xci/server deploy --prod` (D-03) prunes devDeps including drizzle-kit. Verify step asserts `node_modules/drizzle-kit` absent. |
| T-14-01-04 | Information Disclosure | @fastify/static path traversal (e.g. GET /../../etc/passwd) | mitigate | @fastify/static 8.x handles `..` normalization internally. DO NOT set `allowDotFiles: true` (default false). Prefix `/` + root absolute. Covered by @fastify/static test suite upstream; our integration test verifies `GET /../package.json` returns 404 (add as assertion 7 if time permits, else deferred to smoke test Plan 14-03). |
| T-14-01-05 | Denial of Service | Large static file request exhausts bandwidth | accept | Existing rate-limit global 100 req/min covers this; static assets are small (main bundle 177KB gzip per Phase 13); Content-Length bounded by actual files in `/app/web`. No mitigation beyond existing rate-limit. |
| T-14-01-06 | Repudiation | Migrator at boot runs without audit trail | accept | Drizzle migrator writes to __drizzle_migrations table with timestamp; app.log.info emits "running database migrations" / "migrations complete". Sufficient for post-mortem. No audit-log table row needed (not a user action). |
| T-14-01-07 | Elevation of Privilege | Container-escape vulnerability in node:22-slim base | mitigate | Plan 14-03 adds Trivy scan (HIGH/CRITICAL block per D-24). Image rebuilds on each release pull latest base layer patches. |
| T-14-01-08 | Spoofing | HEALTHCHECK curl against wrong endpoint could hide server illness | mitigate | HEALTHCHECK uses /api/healthz (verified route exists in routes/index.ts); returns JSON `{ok:true}` with 200 on success. Curl `-f` flag treats 4xx/5xx as failure. |
