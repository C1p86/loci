<objective>
Stand up the full release pipeline: a tag-triggered GitHub Actions workflow that builds the Dockerfile from Plan 14-01, runs an end-to-end smoke test against the built image (PKG-08), runs Trivy container vulnerability scan (D-24), pushes the image with coordinated tags to ghcr.io (D-17). The smoke test is the real gate for PKG-08. Coordinated tagging (latest / vX.Y.Z / vX.Y / vX) makes the image consumable by SRE teams who pin to various precision levels.

    **Step 1 — Create `scripts/smoke.mjs`** at repo root (ESM, Node 22+ native fetch, no external deps):

    ```javascript
    #!/usr/bin/env node
    // scripts/smoke.mjs
    // PKG-08 — end-to-end smoke against a running xci/server image.
    // Usage: node scripts/smoke.mjs [baseUrl] [mailhogBaseUrl]
    //   defaults: baseUrl=http://localhost:3000, mailhogBaseUrl=http://localhost:8025

    import { randomBytes } from 'node:crypto';

    const BASE = process.argv[2] ?? 'http://localhost:3000';
    const MAILHOG = process.argv[3] ?? 'http://localhost:8025';

    const STEP = (n, msg) => console.log(`[smoke:${n}] ${msg}`);
    const FAIL = (msg) => { console.error(`[smoke:FAIL] ${msg}`); process.exit(1); };

    // --- cookie jar (single-pair, sufficient for this flow) ---
    let sessionCookie = '';
    let csrfCookie = '';
    const cookieHeader = () => [sessionCookie, csrfCookie].filter(Boolean).join('; ');

    function captureCookies(res) {
      const all = typeof res.headers.getSetCookie === 'function'
        ? res.headers.getSetCookie()
        : (res.headers.get('set-cookie') ? res.headers.get('set-cookie').split(/,(?=\s*[\w-]+=)/) : []);
      for (const raw of all) {
        const [kv] = raw.split(';');
        if (kv.startsWith('xci_sid=')) sessionCookie = kv.trim();
        if (kv.startsWith('_csrf=')) csrfCookie = kv.trim();
      }
    }

    async function json(path, { method = 'GET', body, csrf } = {}) {
      const headers = { 'content-type': 'application/json' };
      if (cookieHeader()) headers.cookie = cookieHeader();
      if (csrf) headers['x-csrf-token'] = csrf;
      const res = await fetch(`${BASE}${path}`, {
        method,
        headers,
        body: body ? JSON.stringify(body) : undefined,
      });
      captureCookies(res);
      const text = await res.text();
      let payload; try { payload = text ? JSON.parse(text) : {}; } catch { payload = { raw: text }; }
      return { status: res.status, payload };
    }

    // 1. Healthcheck
    STEP(1, `healthz`);
    {
      const res = await fetch(`${BASE}/api/healthz`);
      if (res.status !== 200) FAIL(`healthz ${res.status}`);
      const body = await res.json();
      if (body.ok !== true) FAIL(`healthz body ${JSON.stringify(body)}`);
    }

    // 2. Signup
    const email = `smoke-${randomBytes(6).toString('hex')}@example.com`;
    const password = `SmokePass-${randomBytes(8).toString('base64url')}!`;
    STEP(2, `signup ${email}`);
    {
      const { status, payload } = await json('/api/auth/signup', {
        method: 'POST',
        body: { email, password, name: 'Smoke Tester' },
      });
      if (status !== 201) FAIL(`signup ${status} ${JSON.stringify(payload)}`);
    }

    // 3. Poll mailhog for verification token
    STEP(3, `mailhog poll for verify token`);
    let verifyToken = null;
    for (let attempt = 0; attempt < 30 && !verifyToken; attempt++) {
      await new Promise((r) => setTimeout(r, 500));
      const mh = await fetch(`${MAILHOG}/api/v2/messages`);
      if (!mh.ok) continue;
      const mj = await mh.json();
      for (const msg of mj.items ?? []) {
        const toMatch = (msg.To ?? []).some(
          (t) => `${t.Mailbox}@${t.Domain}`.toLowerCase() === email.toLowerCase(),
        );
        if (!toMatch) continue;
        const bodyText = msg.Content?.Body ?? '';
        const m = bodyText.match(/token=([A-Za-z0-9_-]+)/);
        if (m) { verifyToken = m[1]; break; }
      }
    }
    if (!verifyToken) FAIL('never received verification email');

    // 4. Verify email
    STEP(4, `verify-email`);
    {
      const { status } = await json('/api/auth/verify-email', { method: 'POST', body: { token: verifyToken } });
      if (status !== 200) FAIL(`verify ${status}`);
    }

    // 5. Login
    STEP(5, `login`);
    {
      const { status, payload } = await json('/api/auth/login', { method: 'POST', body: { email, password } });
      if (status !== 200) FAIL(`login ${status} ${JSON.stringify(payload)}`);
    }

    // 6. /auth/me — get active orgId
    STEP(6, `auth/me`);
    const me = await json('/api/auth/me');
    if (me.status !== 200) FAIL(`me ${me.status}`);
    const orgId = me.payload.activeOrgId ?? me.payload.orgId ?? me.payload.org?.id ?? me.payload.org_id;
    if (!orgId) FAIL(`no orgId in /auth/me: ${JSON.stringify(me.payload)}`);

    // 7. Fetch CSRF token
    STEP(7, `csrf`);
    const csrfRes = await json('/api/auth/csrf');
    if (csrfRes.status !== 200) FAIL(`csrf ${csrfRes.status}`);
    const csrf = csrfRes.payload.token;
    if (!csrf) FAIL('no csrf token in response');

    // 8. Create registration token
    STEP(8, `registration token`);
    {
      const { status, payload } = await json(
        `/api/orgs/${orgId}/agents/registration-tokens`,
        { method: 'POST', body: {}, csrf },
      );
      if (status !== 201) FAIL(`regtoken ${status} ${JSON.stringify(payload)}`);
      if (!payload.token) FAIL('regtoken missing token');
    }

    // 9. Create task
    STEP(9, `create task`);
    let taskId;
    {
      const { status, payload } = await json(`/api/orgs/${orgId}/tasks`, {
        method: 'POST',
        csrf,
        body: {
          name: 'Smoke Task',
          slug: 'smoke-task',
          yaml_definition: 'default:\n  echo: ["echo", "smoke"]\n',
          label_requirements: [],
        },
      });
      if (status !== 201) FAIL(`task create ${status} ${JSON.stringify(payload)}`);
      taskId = payload.id ?? payload.task_id;
      if (!taskId) FAIL('no taskId');
    }

    // 10. Trigger run
    STEP(10, `trigger run`);
    let runId;
    {
      const { status, payload } = await json(
        `/api/orgs/${orgId}/tasks/${taskId}/runs`,
        { method: 'POST', csrf, body: {} },
      );
      if (status !== 202) FAIL(`trigger ${status} ${JSON.stringify(payload)}`);
      runId = payload.id ?? payload.run_id;
      if (!runId) FAIL('no runId');
    }

    // 11. Fetch run state (should be queued/dispatched/running — no agent connected, stays queued)
    STEP(11, `run state`);
    {
      const { status, payload } = await json(`/api/orgs/${orgId}/runs/${runId}`);
      if (status !== 200) FAIL(`run get ${status}`);
      const ok = ['queued', 'dispatched', 'running'].includes(payload.status);
      if (!ok) FAIL(`run status unexpected: ${payload.status}`);
    }

    // 12. Usage endpoint
    STEP(12, `usage`);
    {
      const { status, payload } = await json(`/api/orgs/${orgId}/usage`);
      if (status !== 200) FAIL(`usage ${status}`);
      if (payload.agents?.limit !== 5) FAIL(`usage.agents.limit ${payload.agents?.limit}`);
    }

    // 13. SPA index (PKG-05)
    STEP(13, `SPA index`);
    {
      const res = await fetch(`${BASE}/`);
      if (res.status !== 200) FAIL(`spa index ${res.status}`);
      const body = await res.text();
      if (!body.toLowerCase().includes('<!doctype')) FAIL('spa index not html');
    }

    console.log('[smoke:PASS] all 13 steps green');
    ```

    **Rationale**:
    - Node 22+ native `fetch` + `getSetCookie()` (fallback regex split for older shims); zero npm deps — survives in CI without pnpm install.
    - Agent dispatch E2E deliberately out of scope (per CONTEXT D-12 simplification); queued-state assertion is sufficient evidence for PKG-08. Agent round-trip is already covered by Phase 10-05 / Phase 11-04 integration tests in ci.yml.
    - Uses MailHog HTTP API (`/api/v2/messages`) — matches dev compose pattern from Plan 14-02.
    - If `/api/auth/me` response shape evolves, payload.activeOrgId / .orgId / .org.id / .org_id are all checked.

    ```yaml
    name: Docker Release

    on:
      push:
        tags:
          - 'v*.*.*'            # vX.Y.Z and vX.Y.Z-rc.N all match
      workflow_dispatch:
        inputs:
          tag:
            description: 'Tag to build (e.g., v2.0.0-rc.1); defaults to current ref'
            required: false

    concurrency:
      group: docker-release-${{ github.ref }}
      cancel-in-progress: false

    jobs:
      build-smoke-scan-push:
        runs-on: ubuntu-latest
        timeout-minutes: 30
        permissions:
          contents: read        # checkout
          packages: write       # push to ghcr.io
          id-token: write       # (future: cosign keyless signing, out of scope v2.0)

        steps:
          - name: Checkout
            uses: actions/checkout@v4

          - name: Determine image tags
            id: tags
            run: |
              REF="${{ github.event.inputs.tag || github.ref_name }}"
              echo "ref=$REF" >> $GITHUB_OUTPUT
              # Strip leading 'v' to get semver
              VER="${REF#v}"
              MAJOR=$(echo "$VER" | cut -d. -f1)
              MINOR=$(echo "$VER" | cut -d. -f1-2)
              # Pre-release (e.g. v2.0.0-rc.1) → only push versioned tag, skip latest/major/minor floating tags
              if [[ "$VER" == *-* ]]; then
                echo "is_prerelease=true" >> $GITHUB_OUTPUT
              else
                echo "is_prerelease=false" >> $GITHUB_OUTPUT
              fi
              echo "version=$VER" >> $GITHUB_OUTPUT
              echo "major=v$MAJOR" >> $GITHUB_OUTPUT
              echo "minor=v$MINOR" >> $GITHUB_OUTPUT

          - name: Setup Docker Buildx
            uses: docker/setup-buildx-action@v3

          - name: Log in to ghcr.io
            uses: docker/login-action@v3
            with:
              registry: ghcr.io
              username: ${{ github.actor }}
              password: ${{ secrets.GITHUB_TOKEN }}

          # Build image locally (no push yet) — we smoke + scan first
          - name: Build image (load into local Docker daemon)
            uses: docker/build-push-action@v6
            with:
              context: .
              file: packages/server/Dockerfile
              platforms: linux/amd64           # D-18: ARM64 deferred
              load: true                       # load into local daemon for smoke + scan
              push: false
              tags: |
                xci-server:smoke-candidate
              cache-from: type=gha
              cache-to: type=gha,mode=max

          - name: Verify image constraints (size, non-root, drizzle-kit absent)
            run: |
              SIZE=$(docker image inspect xci-server:smoke-candidate --format '{{.Size}}')
              echo "image size bytes: $SIZE"
              if [ "$SIZE" -gt 419430400 ]; then
                echo "FAIL: image ${SIZE} bytes exceeds 400MB"
                exit 1
              fi
              UID=$(docker run --rm xci-server:smoke-candidate id -u)
              if [ "$UID" != "10001" ]; then
                echo "FAIL: container UID is $UID (expected 10001)"
                exit 1
              fi
              if docker run --rm xci-server:smoke-candidate ls node_modules/drizzle-kit 2>/dev/null; then
                echo "FAIL: drizzle-kit is present in runtime image"
                exit 1
              fi
              echo "PASS: size/uid/devdep-prune checks green"

          # Smoke test — spin up ephemeral postgres + mailhog + image, run scripts/smoke.mjs
          - name: Create test .env for smoke
            run: cp .env.example .env

          - name: Start smoke stack with built image
            run: |
              # Override compose to use the pre-built image tag rather than rebuilding
              export XCI_SERVER_IMAGE=xci-server:smoke-candidate
              docker compose up -d postgres mailhog
              # start server manually so we can use the pre-built image (compose would rebuild)
              docker run -d \
                --name xci-smoke-server \
                --network loci_default \
                -e DATABASE_URL=postgres://xci:xci_dev_password@postgres:5432/xci \
                -e SMTP_HOST=mailhog -e SMTP_PORT=1025 \
                -e EMAIL_TRANSPORT=smtp -e SMTP_FROM=xci-dev@localhost \
                -e SESSION_COOKIE_SECRET=smoke-session-secret-at-least-32-chars-xxx \
                -e XCI_MASTER_KEY=AAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA \
                -e PLATFORM_ADMIN_EMAIL=admin@localhost \
                -e NODE_ENV=production \
                -p 3000:3000 \
                -p 8025:8025 \
                xci-server:smoke-candidate
              # Wait for healthy
              for i in {1..40}; do
                HEALTH=$(docker inspect -f '{{.State.Health.Status}}' xci-smoke-server 2>/dev/null || echo starting)
                echo "attempt $i: health=$HEALTH"
                [ "$HEALTH" = "healthy" ] && break
                sleep 3
              done
              [ "$HEALTH" = "healthy" ] || { docker logs xci-smoke-server; exit 1; }

          - name: Run smoke.mjs
            run: node scripts/smoke.mjs http://localhost:3000 http://localhost:8025

          - name: Dump server logs (always, for post-mortem)
            if: always()
            run: docker logs xci-smoke-server || true

          - name: Tear down smoke stack
            if: always()
            run: |
              docker stop xci-smoke-server || true
              docker rm xci-smoke-server || true
              docker compose down -v || true

          # Container vulnerability scan (D-24)
          - name: Trivy scan (block on HIGH/CRITICAL)
            uses: aquasecurity/trivy-action@0.24.0
            with:
              image-ref: xci-server:smoke-candidate
              format: table
              severity: HIGH,CRITICAL
              exit-code: '1'
              ignore-unfixed: true       # ignore vulns with no upstream fix available yet

          # Push to ghcr.io with coordinated tags (D-17)
          - name: Tag and push to ghcr.io
            run: |
              OWNER_LOWER=$(echo "${{ github.repository_owner }}" | tr '[:upper:]' '[:lower:]')
              BASE="ghcr.io/${OWNER_LOWER}/xci-server"
              # Versioned tag (always)
              docker tag xci-server:smoke-candidate "$BASE:${{ steps.tags.outputs.ref }}"
              docker push "$BASE:${{ steps.tags.outputs.ref }}"
              # Floating tags only on stable release (not pre-release)
              if [ "${{ steps.tags.outputs.is_prerelease }}" = "false" ]; then
                docker tag xci-server:smoke-candidate "$BASE:latest"
                docker tag xci-server:smoke-candidate "$BASE:${{ steps.tags.outputs.minor }}"
                docker tag xci-server:smoke-candidate "$BASE:${{ steps.tags.outputs.major }}"
                docker push "$BASE:latest"
                docker push "$BASE:${{ steps.tags.outputs.minor }}"
                docker push "$BASE:${{ steps.tags.outputs.major }}"
              fi
              echo "Pushed:"
              docker images --format '{{.Repository}}:{{.Tag}}' | grep "$BASE" || true
    ```

    **Constraints — do NOT deviate**:
    - Trigger: `push.tags: ['v*.*.*']` (matches Changesets-emitted tags)
    - `platforms: linux/amd64` only (D-18 defers ARM64)
    - Smoke runs BEFORE Trivy BEFORE push — any gate failing aborts before ghcr.io write
    - Pre-release tags (`-rc.N`) get versioned push only, no `latest` / `major` / `minor` moves
    - `permissions.packages: write` needed for ghcr.io push
    - `ignore-unfixed: true` on Trivy — prevents false-positive churn on vulns with no fix available
    - Compose's `default` network name is `<dirname>_default` — `loci_default` per the repo folder name. If the repo is renamed, this step needs revisit (documented in runbook Plan 14-04)
    - The override-compose approach in "Start smoke stack" — we manually run the pre-built image on the compose network rather than rebuilding via compose. This verifies the IMAGE, not a rebuild. (D-30)
