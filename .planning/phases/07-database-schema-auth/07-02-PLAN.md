```typescript
import { drizzle } from 'drizzle-orm/postgres-js';
import { migrate } from 'drizzle-orm/postgres-js/migrator';
import postgres from 'postgres';
import path from 'node:path';
import { fileURLToPath } from 'node:url';

const __filename = fileURLToPath(import.meta.url);
const __dirname = path.dirname(__filename);

/**
 * D-28: Programmatic migrator at boot. Used both in production server.ts
 * and in test-utils/db-harness.ts after the testcontainer starts.
 * Uses max:1 to avoid holding a pool during boot.
 */
export async function runMigrations(databaseUrl: string): Promise<void> {
  const migrationClient = postgres(databaseUrl, { max: 1 });
  const migrationDb = drizzle(migrationClient);
  try {
    await migrate(migrationDb, {
      // Dev: src/db/migrator.ts — two levels up from __dirname (src/db) to packages/server, then /drizzle
      // Built: dist/db/migrator.js — same relative path to packages/server/drizzle (drizzle ships in package `files`)
      migrationsFolder: path.resolve(__dirname, '..', '..', 'drizzle'),
    });
  } finally {
    await migrationClient.end({ timeout: 5 });
  }
}
```

**Create `packages/server/src/db/plugin.ts`** — fastify-plugin exposing `fastify.db`:
```typescript
import fp from 'fastify-plugin';
import type { FastifyPluginAsync } from 'fastify';
import { drizzle, type PostgresJsDatabase } from 'drizzle-orm/postgres-js';
import postgres from 'postgres';

declare module 'fastify' {
  interface FastifyInstance {
    db: PostgresJsDatabase;
  }
}

const dbPlugin: FastifyPluginAsync<{ databaseUrl?: string }> = async (fastify, opts) => {
  // Read URL from env (registered by fastify-env, available via fastify.config)
  // or from explicit opts for test injection (D-05).
  const url = opts.databaseUrl ?? fastify.config?.DATABASE_URL;
  if (!url) throw new Error('dbPlugin: DATABASE_URL missing from both opts and fastify.config');
  const client = postgres(url, { max: 10 });
  const db = drizzle(client);
  fastify.decorate('db', db);
  fastify.addHook('onClose', async () => {
    await client.end({ timeout: 5 });
  });
};

export default fp(dbPlugin, { name: 'db-plugin', dependencies: ['@fastify/env'] });
```

Note: `dependencies: ['@fastify/env']` enforces registration order (D-06) — fastify throws if db plugin registers before env plugin. `fp(...)` wraps so `fastify.db` is visible outside this plugin scope.
