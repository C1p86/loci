    import { RoleInsufficientError, SessionRequiredError } from '../errors.js';

    // Phase 9 D-24: platform admin gate. Single trusted operator identified by
    // PLATFORM_ADMIN_EMAIL env var. Per-org Owners/Members cannot reach this route.
    // Case-insensitive compare — both sides lowercased.
    export async function requirePlatformAdmin(req: FastifyRequest): Promise<void> {
      if (!req.user) throw new SessionRequiredError();
      const configured = req.server.config.PLATFORM_ADMIN_EMAIL.toLowerCase();
      if (req.user.email.toLowerCase() !== configured) {
        throw new RoleInsufficientError('owner');
      }
    }

    import type { FastifyPluginAsync } from 'fastify';
    import { MekRotationError } from '../../errors.js';
    import { requirePlatformAdmin } from '../../plugins/require-platform-admin.js';
    import { makeRepos } from '../../repos/index.js';

    interface RotateMekBody {
      newMekBase64: string;
    }

    export const rotateMekRoute: FastifyPluginAsync = async (fastify) => {
      fastify.post<{ Body: RotateMekBody }>(
        '/rotate-mek',
        {
          onRequest: [fastify.csrfProtection],
          preHandler: [fastify.requireAuth, requirePlatformAdmin],
          schema: {
            body: {
              type: 'object',
              required: ['newMekBase64'],
              additionalProperties: false,
              properties: {
                newMekBase64: {
                  type: 'string',
                  minLength: 44,
                  maxLength: 44,
                  pattern: '^[A-Za-z0-9+/]{43}=$',
                },
              },
            },
          },
        },
        async (req, reply) => {
          const newMek = Buffer.from(req.body.newMekBase64, 'base64');
          if (newMek.length !== 32) {
            throw new MekRotationError('new MEK must decode to exactly 32 bytes');
          }
          const oldMek = fastify.mek;
          const repos = makeRepos(fastify.db, fastify.mek);
          const result = await repos.admin.rotateMek(oldMek, newMek);
          return reply.status(200).send(result);
        },
      );
    };
    ```
    NOTE on mek-swap semantics: this plan does NOT swap `fastify.mek` to the new key after rotation.     // Phase 9 D-33 + D-34: pure function Phase 10 dispatcher will call.
    // Merges runOverrides (UI per-run override values) + orgSecrets (decrypted at dispatch time)
    // with precedence runOverrides > orgSecrets > unresolved (left for agent-side .xci/secrets.yml).
    // No DB access; no side effects.

    import type { Task } from '../db/schema.js';

    export interface ResolveInput {
      task: Pick<Task, 'yamlDefinition' | 'name' | 'id'>;
      runOverrides: Record<string, string>;
      orgSecrets: Record<string, string>;
    }

    export interface ResolveOutput {
      // The yamlDefinition with placeholders substituted where possible; unknown left as ${VAR}.
      resolvedYaml: string;
      // Names of ${VAR} placeholders present in the YAML that were NOT satisfied by
      // runOverrides or orgSecrets — the agent will try to resolve these from .xci/secrets.yml.
      unresolved: readonly string[];
    }

    const PLACEHOLDER_RE = /\$\{([^}]+)\}/g;

    export function resolveTaskParams(input: ResolveInput): ResolveOutput {
      // D-34: runOverrides wins over orgSecrets on collision
      const merged: Record<string, string> = { ...input.orgSecrets, ...input.runOverrides };
      const unresolved = new Set<string>();
      const resolvedYaml = input.task.yamlDefinition.replace(PLACEHOLDER_RE, (match, key: string) => {
        const name = key.trim();
        if (Object.hasOwn(merged, name)) {
          return merged[name];
        }
        unresolved.add(name);
        return match;
      });
      return { resolvedYaml, unresolved: [...unresolved] };
    }
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| rotate-mek request body newMekBase64 -> Buffer MEK | AJV pattern + base64 decode length check; base64 decode itself has no security-relevant attacks on length-verified input. |
| PLATFORM_ADMIN_EMAIL env var -> requirePlatformAdmin guard | Single-operator trust anchor. Compromise of this email account grants full MEK rotation authority. Documented in runbook. |
| OLD MEK -> NEW MEK transition window | During rotation, both keys may exist in server memory (fastify.mek = OLD; newMek from body). After rotation, a restart loads NEW as fastify.mek. Brief in-memory coexistence is unavoidable for the re-wrap operation. |
| Plaintext invariance across rotation | D-26 acceptance: plaintexts must match pre/post rotation. Verified by the integration test that rebuilds the app with the new mek and calls resolveByName on every secret. |
| dispatch-resolver.ts pure function -> Phase 10 dispatcher | Service is side-effect-free (no DB, no log); contract stability for Phase 10 is the primary concern. |
