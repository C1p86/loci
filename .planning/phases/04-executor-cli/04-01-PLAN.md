**1. Create `src/executor/single.ts`:**

```typescript
import { execa } from 'execa';
import { SpawnError } from '../errors.js';
import type { ExecutionResult } from '../types.js';

export async function runSingle(
  argv: readonly string[],
  cwd: string,
  env: Record<string, string>,
): Promise<ExecutionResult> {
  const [cmd, ...args] = argv;
  if (!cmd) throw new SpawnError('(empty command)', new Error('argv is empty'));
  try {
    const result = await execa(cmd, args, {
      cwd,
      env: { ...process.env, ...env },
      stdout: 'inherit',
      stderr: 'inherit',
      reject: false,
    });
    return { exitCode: result.exitCode ?? 1 };
  } catch (err: unknown) {
    throw new SpawnError(cmd, err);
  }
}
```

Key points: `reject: false` so non-zero exit doesn't throw (EXE-03). Spawn errors (ENOENT) still throw and are caught and wrapped as `SpawnError`. `stdout: 'inherit'` and `stderr: 'inherit'` for real-time streaming (EXE-02). No prefix for single commands (D-07).

**3. Create `src/executor/parallel.ts`:**

```typescript
import { execa } from 'execa';
import { SpawnError } from '../errors.js';
import type { ExecutionResult } from '../types.js';
import { makeLineTransform, printParallelSummary } from './output.js';

export async function runParallel(
  group: readonly { readonly alias: string; readonly argv: readonly string[] }[],
  failMode: 'fast' | 'complete',
  cwd: string,
  env: Record<string, string>,
): Promise<ExecutionResult> {
  const controller = new AbortController();
  const { signal } = controller;

  let wasInterrupted = false;
  const sigintHandler = () => {
    wasInterrupted = true;
    controller.abort(new Error('SIGINT'));
  };
  process.on('SIGINT', sigintHandler);

  const mergedEnv = { ...process.env, ...env };
  const promises = group.map(({ alias, argv }) => {
    const [cmd, ...args] = argv;
    if (!cmd) return Promise.reject(new SpawnError('(empty command)', new Error('argv is empty')));
    return execa(cmd, args, {
      cwd,
      env: mergedEnv,
      stdout: [makeLineTransform(alias), 'inherit'],
      stderr: [makeLineTransform(alias), 'inherit'],
      cancelSignal: signal,
      forceKillAfterDelay: 3000,
      reject: false,
    });
  });

  let firstFailCode = 0;
  const summaryResults: { exitCode: number; canceled: boolean }[] = [];

  if (failMode === 'fast') {
    // Use Promise.allSettled — when a failure is detected, abort remaining
    const settled = await Promise.allSettled(promises);
    process.off('SIGINT', sigintHandler);

    if (wasInterrupted) {
      printParallelSummary(group, settled.map(() => ({ exitCode: 130, canceled: true })));
      return { exitCode: 130 };
    }

    for (const result of settled) {
      if (result.status === 'fulfilled') {
        const code = result.value.exitCode ?? 0;
        summaryResults.push({ exitCode: code, canceled: false });
        if (code !== 0 && firstFailCode === 0) {
          firstFailCode = code;
          controller.abort(new Error('fail-fast'));
        }
      } else {
        const reason = result.reason as { isCanceled?: boolean; exitCode?: number };
        if (reason?.isCanceled) {
          summaryResults.push({ exitCode: 0, canceled: true });
        } else {
          const code = reason?.exitCode ?? 1;
          summaryResults.push({ exitCode: code, canceled: false });
          if (firstFailCode === 0) firstFailCode = code;
        }
      }
    }
  } else {
    // failMode === 'complete': let all finish
    const settled = await Promise.allSettled(promises);
    process.off('SIGINT', sigintHandler);

    if (wasInterrupted) {
      printParallelSummary(group, settled.map(() => ({ exitCode: 130, canceled: true })));
      return { exitCode: 130 };
    }

    for (const result of settled) {
      if (result.status === 'fulfilled') {
        const code = result.value.exitCode ?? 0;
        summaryResults.push({ exitCode: code, canceled: false });
        if (code !== 0 && firstFailCode === 0) firstFailCode = code;
      } else {
        const reason = result.reason as { exitCode?: number };
        const code = reason?.exitCode ?? 1;
        summaryResults.push({ exitCode: code, canceled: false });
        if (firstFailCode === 0) firstFailCode = code;
      }
    }
  }

  printParallelSummary(group, summaryResults);
  return { exitCode: firstFailCode };
}
```

NOTE: The fail-fast abort happens DURING the `Promise.allSettled` iteration. Because `Promise.allSettled` waits for ALL promises, the abort kills remaining children quickly (they receive SIGTERM). The code iterates results in order, so the first non-zero encountered triggers the abort. This means the second failing child may have already completed before the abort — which is fine.
