```yaml
  fence-gates:
    # Linux-only job that enforces the 3-layer ws fence, bundle-size gate, and cold-start gate.
    # Runs in parallel with build-test-lint. Both jobs MUST be required checks on main branch.
    runs-on: ubuntu-latest
    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup pnpm
        uses: pnpm/action-setup@v4

      - name: Setup Node.js 22
        uses: actions/setup-node@v4
        with:
          node-version: 22
          cache: 'pnpm'

      - name: Install dependencies
        run: pnpm install --frozen-lockfile

      - name: Build xci (fresh, for accurate size measurement)
        run: pnpm --filter xci build

      - name: Bundle-size gate (D-15) - dist/cli.mjs under 200 KB
        run: pnpm --filter xci size-limit

      - name: WS-exclusion grep gate (D-16b)
        run: |
          if grep -E "(reconnecting-websocket|['\"]ws['\"])" packages/xci/dist/cli.mjs; then
            echo "::error::Found ws/reconnecting-websocket in dist/cli.mjs - Phase 6 D-16 fence broken!"
            exit 1
          fi
          echo "OK: no ws/reconnecting-websocket strings in dist/cli.mjs"

      - name: Install hyperfine
        run: sudo apt-get update && sudo apt-get install -y hyperfine

      - name: Cold-start gate (D-17) - xci --version mean under 300ms on Linux
        run: |
          hyperfine --runs 10 --warmup 3 \
            --export-json hyperfine.json \
            'node packages/xci/dist/cli.mjs --version'
          node -e "
            const r = require('./hyperfine.json');
            const meanMs = r.results[0].mean * 1000;
            console.log('Mean cold-start:', meanMs.toFixed(1) + 'ms');
            if (meanMs >= 300) {
              console.error('FAIL: mean ' + meanMs.toFixed(1) + 'ms >= 300ms budget');
              process.exit(1);
            }
            console.log('PASS: under 300ms budget');
          "

      - name: Upload hyperfine JSON artifact (for post-mortem)
        if: always()
        uses: actions/upload-artifact@v4
        with:
          name: hyperfine-results
          path: hyperfine.json
```

```yaml
name: Release

on:
  push:
    branches: [main]

# Concurrency ensures we do not run two release attempts at once (e.g. two merges in quick succession)
concurrency: ${{ github.workflow }}-${{ github.ref }}

jobs:
  release:
    runs-on: ubuntu-latest
    # Job-level permissions (least-privilege, explicit grant per RESEARCH.md Pitfall 4).
    # Required by changesets/action@v1:
    #   contents: write        push the Version PR branch
    #   pull-requests: write   create/update the Version PR
    permissions:
      contents: write
      pull-requests: write
    steps:
      - name: Checkout
        uses: actions/checkout@v4
        with:
          # Changesets needs full history to generate changelog entries since the last release
          fetch-depth: 0

      - name: Setup pnpm
        uses: pnpm/action-setup@v4

      - name: Setup Node.js 22
        uses: actions/setup-node@v4
        with:
          node-version: 22
          cache: 'pnpm'

      - name: Install dependencies
        run: pnpm install --frozen-lockfile

      - name: Build all packages (required before publish)
        run: pnpm turbo run build

      - name: Create Release Pull Request or Publish
        uses: changesets/action@v1
        with:
          # version input runs during Version PR creation/update. Default `changeset version` is fine.
          version: pnpm changeset version
          # publish input runs when Version PR merges. Calls the release script in root package.json.
          # The release script is: pnpm -r publish --access=public (per D-11 / D-13)
          publish: pnpm release
        env:
          # Auto-provided by GitHub Actions; needed for changesets to interact with GH API
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}
          # Must be added to repo secrets before first publish (Phase 14+).
          # Phase 6 workflow runs without triggering publish (no changesets authored yet),
          # but the env var must be declared so the action is wired correctly for later.
          NPM_TOKEN: ${{ secrets.NPM_TOKEN }}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| GitHub Actions runner -> pnpm store / npm registry | Untrusted transitive dep install; mitigated by frozen-lockfile and packageManager pin |
| release workflow runner -> npm publish | NPM_TOKEN authenticates publish; token scope must be limited to xci + @xci/server + @xci/web |
| changesets/action@v1 -> GitHub API | Uses GITHUB_TOKEN to push branches and open PRs; permissions must be minimum-viable (contents: write, pull-requests: write only) |
| CI gates -> branch protection | If ci.yml checks are not marked as required, a PR could merge without passing; documented as post-merge action item |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-22 | Tampering (supply chain) | pnpm install in CI | mitigate | `pnpm install --frozen-lockfile` in all three places (build-test-lint job, fence-gates job, release job). Any drift in pnpm-lock.yaml fails the install. packageManager pin in root package.json ensures pnpm version matches across runners. |
| T-06-23 | Information Disclosure | NPM_TOKEN in release.yml | mitigate | Token injected only via `${{ secrets.NPM_TOKEN }}`, never echoed in run commands. Token scope must be publish-only for the three package names (documented as post-merge action item for the user). Never logged. Only used in release.yml (not ci.yml). |
| T-06-24 | Elevation of Privilege | Over-broad workflow permissions | mitigate | release.yml sets explicit `permissions: { contents: write, pull-requests: write }` at workflow level. Default GITHUB_TOKEN permissions do NOT include these; making the grant explicit is the principle-of-least-privilege pattern. ci.yml does NOT need write permissions (read-only reads + apt install in Linux runner which is container-sandboxed). |
| T-06-25 | Denial of Service | hyperfine gate flake due to runner variance (Pitfall 3) | accept | Threshold is 300ms; with --runs 10 --warmup 3, expected mean on current 126 KB bundle is ~200ms with 5-10% variance.
